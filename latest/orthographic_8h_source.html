21 /// Returns orthographic camera model given bounding box and image size.
22 template <class TScalar>
23 OrthographicModelT<TScalar> orthoCamFromBoundingBox(
24  Region2<TScalar> const& bounding_box, ImageSize image_size) {
25  // (-0.5, -0.5) -> (min.x, min.y)
26  // (-0.5, h-0.5) -> (min.x, max.y)
27  // (w-0.5, -0.5) -> (max.x, min.y)
28  // (w-0.5, h-0.5) -> (max.x, max.y)
29  //
30  // Thus we have the following relationship:
31  // u = x*sx + ox
32  // v = y*sy + oy
33  //
34  // -0.5 = min.x * sx + ox => ox = -(0.5 + min.x * sx)
35  // (w-0.5) = max.x * sx + ox
36  //
37  // w = sx * (max.x - min.x)
38  // sx = w / (max.x - min.x)
39 
40  Eigen::Array<TScalar, 2, 1> const range = bounding_box.range();
41  Eigen::Array<TScalar, 2, 1> const scale =
42  Eigen::Array<TScalar, 2, 1>(image_size.width, image_size.height) / range;
43  Eigen::Array<TScalar, 2, 1> const offset =
44  -(0.5 + bounding_box.min().array() * scale);
45 
46  Eigen::Matrix<TScalar, 4, 1> const params(
47  scale.x(), scale.y(), offset.x(), offset.y());
48 
49  return OrthographicModelT<TScalar>(image_size, params);
50 }

52 /// Returns 2d bounding box corresponding the the given orthographic camera
53 /// model.
54 template <class TScalar>
55 Region2<TScalar> boundingBoxFromOrthoCam(
56  OrthographicModelT<TScalar> const& ortho_cam) {
57  Eigen::Vector<TScalar, 2> min = (-ortho_cam.principalPoint().array() - 0.5) /
58  ortho_cam.focalLength().array();
59  return Region2<TScalar>::fromMinMax(
60  min,
61  min.array() + ortho_cam.imageSize().array().template cast<TScalar>() /
62  ortho_cam.focalLength().array());
63 }
