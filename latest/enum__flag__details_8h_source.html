140 #define FARM_ENUMFLAGS_DEF_IMPL(NAME, UINT_TYPE, ...) \
141  namespace enum_wrapper_ { \
142  \
143  static_assert( \
144  std::is_unsigned<UINT_TYPE>::value, \
145  FARM_PP_STRINGIZE(UINT_TYPE) " must be an unsigned integer type"); \
146  \
147  enum class NAME##Impl : UINT_TYPE{ \
148  none = 0u, FARM_ENUMFLAGS_DETAILS_FLAG_DEFINITIONS(__VA_ARGS__)}; \
149  \
150  NAME##Impl inline constexpr operator|(NAME##Impl left, NAME##Impl right) { \
151  return NAME##Impl(UINT_TYPE(left) | UINT_TYPE(right)); \
152  } \
153  \
154  NAME##Impl inline constexpr operator&(NAME##Impl left, NAME##Impl right) { \
155  return NAME##Impl(UINT_TYPE(left) & UINT_TYPE(right)); \
156  } \
157  \
158  [[maybe_unused]] inline bool trySetFlagFromString( \
159  NAME##Impl &value, std::string const &str) { \
160  FARM_PP_SEQ_FOR_EACH( \
161  FARM_ENUMFLAG_DETAILS_OP_SET_ENUM_FROM_STRING, \
162  NAME##Impl, \
163  FARM_PP_TUPLE_TO_SEQ(__VA_ARGS__)) \
164  return false; \
165  } \
166  \
167  [[maybe_unused]] inline bool constexpr hasMask( \
168  NAME##Impl value, NAME##Impl mask) { \
169  return (value & mask) == mask; \
170  } \
171  \
172  [[maybe_unused]] inline void setMask(NAME##Impl &value, NAME##Impl mask) { \
173  value = value | mask; \
174  } \
175  \
176  [[maybe_unused]] inline void clearMask(NAME##Impl &value, NAME##Impl mask) { \
177  value = value & NAME##Impl(~UINT_TYPE(mask)); \
178  } \
179  \
180  [[maybe_unused]] inline void toggleMask( \
181  NAME##Impl &value, NAME##Impl mask) { \
182  value = NAME##Impl(UINT_TYPE(value) ^ UINT_TYPE(mask)); \
183  } \
184  \
185  [[maybe_unused]] inline bool constexpr isSingleFlag(NAME##Impl value) { \
186  switch (value) { \
187  FARM_PP_SEQ_FOR_EACH( \
188  FARM_ENUMFLAG_DETAILS_OP_SINGLE_FLAG_CHECK, \
189  NAME##Impl, \
190  FARM_PP_TUPLE_TO_SEQ(__VA_ARGS__)) \
191  case NAME##Impl::none: \
192  break; \
193  } \
194  return false; \
195  } \
196  \
197  [[maybe_unused]] inline std::vector<std::string> toStrings( \
198  NAME##Impl value) { \
199  std::vector<std::string> strings; \
200  FARM_PP_SEQ_FOR_EACH( \
201  FARM_ENUMFLAG_DETAILS_OP_PUSHBACK_STRING, \
202  NAME##Impl, \
203  FARM_PP_TUPLE_TO_SEQ(__VA_ARGS__)) \
204  return strings; \
205  } \
206  \
207  [[maybe_unused]] inline std::string toPretty(NAME##Impl value) { \
208  std::string prettySet("{"); \
209  FARM_PP_SEQ_FOR_EACH( \
210  FARM_ENUMFLAG_DETAILS_OP_STRING_FOR_FLAG, \
211  NAME##Impl, \
212  FARM_PP_TUPLE_TO_SEQ(__VA_ARGS__)) \
213  prettySet += "} (=" + std::to_string(UINT_TYPE(value)) + ")"; \
214  return prettySet; \
215  } \
