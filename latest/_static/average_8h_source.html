29 /// Calculates mean iteratively.
30 ///
31 /// Returns ``nullopt`` if it does not converge.
32 ///
33 template <class TSequenceContainer>
34 std::optional<typename TSequenceContainer::value_type> iterativeMean(
35  TSequenceContainer const& foo_transforms_bar, int max_num_iterations) {
36  size_t k_matrix_dim = foo_transforms_bar.size();
37  SOPHUS_ASSERT(k_matrix_dim >= 1, "kMatrixDim must be >= 1.");
38 
39  using Group = typename TSequenceContainer::value_type;
40  using Scalar = typename Group::Scalar;
41  using Tangent = typename Group::Tangent;
42 
43  // This implements the algorithm in the beginning of Sec. 4.2 in
44  // ftp://ftp-sop.inria.fr/epidaure/Publications/Arsigny/arsigny_rr_biinvariant_average.pdf.
45  Group foo_transform_average = foo_transforms_bar.front();
46  Scalar w = Scalar(1. / k_matrix_dim);
47  for (int i = 0; i < max_num_iterations; ++i) {
48  Tangent average;
49  setToZero<Tangent>(average);
50  for (Group const& foo_transform_bar : foo_transforms_bar) {
51  average +=
52  w * (foo_transform_average.inverse() * foo_transform_bar).log();
53  }
54  Group foo_transform_newaverage =
55  foo_transform_average * Group::exp(average);
56  if (squaredNorm<Tangent>(
57  (foo_transform_newaverage.inverse() * foo_transform_average)
58  .log()) < kEpsilon<Scalar>) {
59  return foo_transform_newaverage;
60  }
61 
62  foo_transform_average = foo_transform_newaverage;
63  }
64  // LCOV_EXCL_START
65  return std::nullopt;
66  // LCOV_EXCL_STOP
67 }

76 // Mean implementation for Cartesian.
77 template <
78  class TSequenceContainer,
79  int kPointDim = TSequenceContainer::value_type::kDoF,
80  class TScalar = typename TSequenceContainer::value_type::Scalar>
81 std::enable_if_t<
82  std::is_same<
83  typename TSequenceContainer::value_type,
84  Cartesian<TScalar, kPointDim> >::value,
85  std::optional<typename TSequenceContainer::value_type> >
86 average(TSequenceContainer const& foo_transforms_bar) {
87  size_t k_matrix_dim = std::distance(
88  std::begin(foo_transforms_bar), std::end(foo_transforms_bar));
89  SOPHUS_ASSERT(k_matrix_dim >= 1, "kMatrixDim must be >= 1.");
90 
91  Eigen::Vector<TScalar, kPointDim> average;
92  average.setZero();
93  for (Cartesian<TScalar, kPointDim> const& foo_transform_bar :
94  foo_transforms_bar) {
95  average += foo_transform_bar.params();
96  }
97  return Cartesian<TScalar, kPointDim>(average / TScalar(k_matrix_dim));
98 }

108  // This implements rotational part of Proposition 12 from Sec. 6.2 of

159 template <
160  class TSequenceContainer,
161  class TScalar = typename TSequenceContainer::value_type::Scalar>
162 Eigen::Quaternion<TScalar> averageUnitQuaternion(
163  TSequenceContainer const& foo_transforms_bar) {
164  // This: http://stackoverflow.com/a/27410865/1221742
165  size_t k_matrix_dim = std::distance(
166  std::begin(foo_transforms_bar), std::end(foo_transforms_bar));
167  SOPHUS_ASSERT(k_matrix_dim >= 1, "kMatrixDim must be >= 1.");
168  Eigen::Matrix<TScalar, 4, Eigen::Dynamic> q(4, k_matrix_dim);
169  int i = 0;
170  TScalar w = TScalar(1. / k_matrix_dim);
171  for (auto const& foo_transform_bar : foo_transforms_bar) {
172  q.col(i) = w * details::getUnitQuaternion(foo_transform_bar).coeffs();
173  ++i;
174  }
175 
176  Eigen::Matrix<TScalar, 4, 4> q_qt = q * q.transpose();
177  // TODO: Figure out why we can't use SelfAdjointEigenSolver here.
178  Eigen::EigenSolver<Eigen::Matrix<TScalar, 4, 4> > es(q_qt);
179 
180  std::complex<TScalar> max_eigenvalue = es.eigenvalues()[0];
181  Eigen::Matrix<std::complex<TScalar>, 4, 1> max_eigenvector =
182  es.eigenvectors().col(0);
183 
184  for (int i = 1; i < 4; i++) {
185  if (std::norm(es.eigenvalues()[i]) > std::norm(max_eigenvalue)) {
186  max_eigenvalue = es.eigenvalues()[i];
187  max_eigenvector = es.eigenvectors().col(i);
188  }
189  }
190  Eigen::Quaternion<TScalar> quat;
191  quat.coeffs() << //
192  max_eigenvector[0].real(), //
193  max_eigenvector[1].real(), //
194  max_eigenvector[2].real(), //
195  max_eigenvector[3].real();
196  return quat;
197 }

214 template <
215  class TSequenceContainer,
216  class TScalar = typename TSequenceContainer::value_type::Scalar>
217 std::enable_if_t<
218  std::is_same<typename TSequenceContainer::value_type, RxSo3<TScalar> >::
219  value,
220  std::optional<typename TSequenceContainer::value_type> >
221 average(TSequenceContainer const& foo_transforms_bar) {
222  size_t k_matrix_dim = std::distance(
223  std::begin(foo_transforms_bar), std::end(foo_transforms_bar));
224 
225  SOPHUS_ASSERT(k_matrix_dim >= 1, "kMatrixDim must be >= 1.");
226  TScalar scale_sum = TScalar(0);
227  using std::exp;
228  using std::log;
229  for (RxSo3<TScalar> const& foo_transform_bar : foo_transforms_bar) {
230  scale_sum += log(foo_transform_bar.scale());
231  }
232  return RxSo3<TScalar>(
233  exp(scale_sum / TScalar(k_matrix_dim)),
234  So3<TScalar>(details::averageUnitQuaternion(foo_transforms_bar)));
235 }
