58 /// Se2 base type - implements Se2 class but is storage agnostic.
59 ///
60 /// SE(2) is the group of rotations and translation in 2d. It is the
61 /// semi-direct product of SO(2) and the 2d Euclidean vector space. The class
62 /// is represented using a composition of SO2Group for rotation and a 2-vector
63 /// for translation.
64 ///
65 /// SE(2) is neither compact, nor a commutative group.
66 ///
67 /// - 3x3 Eigen::Matrix representation:
68 ///
69 /// ```
70 /// | R t |
71 /// | o 1 |
72 /// ```
73 ///
74 /// where ``R`` is a 2x2 rotation matrix, ``t`` a translation 2-vector and
75 /// ``o`` a 2-column vector of zeros.
76 ///
77 /// - Tangent 3-vector: [upsilon, theta],
78 ///
79 /// where ``upsilon`` is the translational velocity 2-vector and ``theta`` the
80 /// rotational velocity.
81 ///
82 /// - Internal 4-vector representation: [t0, t1, zr, zi],
83 ///
84 /// with ``t0, t1`` are the translational components, and ``zr``
85 /// the real part and ``zi`` the imaginary part of a unit length complex number.
86 ///
87 /// See SO2Group for more details of the rotation representation in 2d.
88 ///
89 template <class TDerived>
90 class Se2Base {
91  public:
92  using Scalar = typename Eigen::internal::traits<TDerived>::Scalar;
93  using TranslationType =
94  typename Eigen::internal::traits<TDerived>::TranslationType;
95  using So2Type = typename Eigen::internal::traits<TDerived>::So2Type;

132  /// Adjoint transformation
133  ///
134  /// This function return the adjoint transformation ``Ad`` of the group
135  /// element ``A`` such that for all ``x`` it holds that
136  /// ``hat(Ad_A * x) = A * hat(x) A^{-1}``. See hat-operator below.
137  ///
138  SOPHUS_FUNC [[nodiscard]] Adjoint adj() const {
139  Eigen::Matrix<Scalar, 2, 2> const& mat_r = so2().matrix();
140  Transformation res;
141  res.setIdentity();
142  res.template topLeftCorner<2, 2>() = mat_r;
143  res(0, 2) = translation()[1];
144  res(1, 2) = -translation()[0];
145  return res;
146  }

157  /// Returns derivative of this * exp(x) wrt x at x=0.
158  ///
159  SOPHUS_FUNC [[nodiscard]] Eigen::Matrix<Scalar, kNumParameters, kDoF>
160  dxThisMulExpXAt0() const {
161  Eigen::Matrix<Scalar, kNumParameters, kDoF> dx;
162  Eigen::Vector2<Scalar> const z = unitComplex();
163  Scalar o(0);
164  // clang-format off
165  dx(0, 0) = o; dx(1, 0) = o; dx(2, 0) = z[0]; dx(3, 0) = z[1];
166  dx(0, 1) = o; dx(1, 1) = o; dx(2, 1) = -z[1]; dx(3, 1) = z[0];
167  dx(0, 2) = -z[1]; dx(1, 2) = z[0]; dx(2, 2) = o; dx(3, 2) = o;
168  // clang-format on
169  return dx;
170  }

172  /// Returns derivative of log(this^{-1} * x) by x at x=this.
173  ///
174  SOPHUS_FUNC [[nodiscard]] Eigen::Matrix<Scalar, kDoF, kNumParameters>
175  dxLogThisInvTimesXAtThis() const {
176  Eigen::Matrix<Scalar, kDoF, kNumParameters> d;
177  d.template block<2, 2>(0, 0).setZero();
178  d.template block<2, 2>(0, 2) = so2().inverse().matrix();
179  d.template block<1, 2>(2, 0) = so2().dxLogThisInvTimesXAtThis();
180  d.template block<1, 2>(2, 2).setZero();

191  /// Logarithmic map
192  ///
193  /// Computes the logarithm, the inverse of the group exponential which maps
194  /// element of the group (rigid body transformations) to elements of the
195  /// tangent space (twist).
196  ///
197  /// To be specific, this function computes ``vee(logmat(.))`` with
198  /// ``logmat(.)`` being the matrix logarithm and ``vee(.)`` the vee-operator
199  /// of SE(2).
200  ///
201  SOPHUS_FUNC [[nodiscard]] Tangent log() const {
202  using std::abs;
203 
204  Tangent vec_upsilon_theta;
205  Scalar theta = so2().log();
206  vec_upsilon_theta[2] = theta;
207  Scalar halftheta = Scalar(0.5) * theta;
208  Scalar halftheta_by_tan_of_halftheta;
209 
210  Eigen::Vector2<Scalar> z = so2().unitComplex();
211  Scalar real_minus_one = z.x() - Scalar(1.);
212  if (abs(real_minus_one) < kEpsilon<Scalar>) {
213  halftheta_by_tan_of_halftheta =
214  Scalar(1.) - Scalar(1. / 12) * theta * theta;
215  } else {
216  halftheta_by_tan_of_halftheta = -(halftheta * z.y()) / (real_minus_one);
217  }
218  Eigen::Matrix<Scalar, 2, 2> v_inv;
219  v_inv << halftheta_by_tan_of_halftheta, halftheta, -halftheta,
220  halftheta_by_tan_of_halftheta;
221  vec_upsilon_theta.template head<2>() = v_inv * translation();
222  return vec_upsilon_theta;
223  }

279  /// Group action on 2-points.
280  ///
281  /// This function rotates and translates a two dimensional point ``p`` by the
282  /// SE(2) element ``bar_transform_foo = (bar_R_foo, t_bar)`` (= rigid body
283  /// transformation):
284  ///
285  /// ``p_bar = bar_R_foo * p_foo + t_bar``.
286  ///
287  template <
288  typename TPointDerived,
289  typename = typename std::enable_if<
290  IsFixedSizeVector<TPointDerived, 2>::value>::type>
291  SOPHUS_FUNC PointProduct<TPointDerived> operator*(
292  Eigen::MatrixBase<TPointDerived> const& p) const {
293  return so2() * p + translation();
294  }
295 
296  /// Group action on homogeneous 2-points. See above for more details.
297  ///
298  template <
299  typename THPointDerived,
300  typename = typename std::enable_if<
301  IsFixedSizeVector<THPointDerived, 3>::value>::type>
302  SOPHUS_FUNC HomogeneousPointProduct<THPointDerived> operator*(
303  Eigen::MatrixBase<THPointDerived> const& p) const {
304  PointProduct<THPointDerived> const tp =
305  so2() * p.template head<2>() + p(2) * translation();
306  return HomogeneousPointProduct<THPointDerived>(tp(0), tp(1), p(2));
307  }

309  /// Group action on lines.
310  ///
311  /// This function rotates and translates a parametrized line
312  /// ``l(t) = o + t * d`` by the SE(2) element:
313  ///
314  /// Origin ``o`` is rotated and translated using SE(2) action
315  /// Direction ``d`` is rotated using SO(2) action
316  ///
317  SOPHUS_FUNC Line operator*(Line const& l) const {
318  return Line((*this) * l.origin(), so2() * l.direction());
319  }

321  /// Group action on hyper-planes.
322  ///
323  /// This function rotates a hyper-plane ``n.x + d = 0`` by the Se2
324  /// element:
325  ///
326  /// Normal vector ``n`` is rotated
327  /// Offset ``d`` is adjusted for translation
328  ///
329  /// Note that in 2d-case hyper-planes are just another parametrization of
330  /// lines
331  ///
332  SOPHUS_FUNC Hyperplane operator*(Hyperplane const& p) const {
333  Hyperplane const rotated = so2() * p;
334  return Hyperplane(
335  rotated.normal(),
336  rotated.offset() - translation().dot(rotated.normal()));
337  }

551  /// Returns derivative of exp(x) wrt. x.
552  ///
553  SOPHUS_FUNC static Eigen::Matrix<Scalar, kNumParameters, kDoF> dxExpX(
554  Tangent const& upsilon_theta) {
555  using std::abs;
556  using std::cos;
557  using std::pow;
558  using std::sin;
559  Eigen::Matrix<Scalar, kNumParameters, kDoF> dx;
560  Eigen::Vector<Scalar, 2> upsilon = upsilon_theta.template head<2>();
561  Scalar theta = upsilon_theta[2];
562 
563  if (abs(theta) < kEpsilon<Scalar>) {
564  Scalar const o(0);
565  Scalar const i(1);
566 
567  // clang-format off
568  dx << o, o, o,
569  o, o, i,
570  i, o, -Scalar(0.5) * upsilon[1],
571  o, i, Scalar(0.5) * upsilon[0];
572  // clang-format on
573  return dx;
574  }
575 
576  Scalar const c0 = sin(theta);
577  Scalar const c1 = cos(theta);
578  Scalar const c2 = 1.0 / theta;
579  Scalar const c3 = c0 * c2;
580  Scalar const c4 = -c1 + Scalar(1);
581  Scalar const c5 = c2 * c4;
582  Scalar const c6 = c1 * c2;
583  Scalar const c7 = pow(theta, -2);
584  Scalar const c8 = c0 * c7;
585  Scalar const c9 = c4 * c7;
586 
587  Scalar const o = Scalar(0);
588  dx(0, 0) = o;
589  dx(0, 1) = o;
590  dx(0, 2) = -c0;
591 
592  dx(1, 0) = o;
593  dx(1, 1) = o;
594  dx(1, 2) = c1;
595 
596  dx(2, 0) = c3;
597  dx(2, 1) = -c5;
598  dx(2, 2) =
599  -c3 * upsilon[1] + c6 * upsilon[0] - c8 * upsilon[0] + c9 * upsilon[1];
600 
601  dx(3, 0) = c5;
602  dx(3, 1) = c3;
603  dx(3, 2) =
604  c3 * upsilon[0] + c6 * upsilon[1] - c8 * upsilon[1] - c9 * upsilon[0];
605  return dx;
606  }

641  /// Group exponential
642  ///
643  /// This functions takes in an element of tangent space (= twist ``a``) and
644  /// returns the corresponding element of the group SE(2).
645  ///
646  /// The first two components of ``a`` represent the translational part
647  /// ``upsilon`` in the tangent space of SE(2), while the last three components
648  /// of ``a`` represents the rotation vector ``omega``.
649  /// To be more specific, this function computes ``expmat(hat(a))`` with
650  /// ``expmat(.)`` being the matrix exponential and ``hat(.)`` the hat-operator
651  /// of SE(2), see below.
652  ///
653  SOPHUS_FUNC static Se2<Scalar> exp(Tangent const& vec_a) {
654  Scalar theta = vec_a[2];
655  So2<Scalar> so2 = So2<Scalar>::exp(theta);
656  Scalar sin_theta_by_theta;
657  Scalar one_minus_cos_theta_by_theta;
658  using std::abs;
659 
660  if (abs(theta) < kEpsilon<Scalar>) {
661  Scalar theta_sq = theta * theta;
662  sin_theta_by_theta = Scalar(1.) - Scalar(1. / 6.) * theta_sq;
663  one_minus_cos_theta_by_theta =
664  Scalar(0.5) * theta - Scalar(1. / 24.) * theta * theta_sq;
665  } else {
666  sin_theta_by_theta = so2.unitComplex().y() / theta;
667  one_minus_cos_theta_by_theta =
668  (Scalar(1.) - so2.unitComplex().x()) / theta;
669  }
670  Eigen::Vector2<Scalar> trans(
671  sin_theta_by_theta * vec_a[0] - one_minus_cos_theta_by_theta * vec_a[1],
672  one_minus_cos_theta_by_theta * vec_a[0] +
673  sin_theta_by_theta * vec_a[1]);
674  return Se2<Scalar>(so2, trans);
675  }
676 
677  /// Returns closest Se3 given arbitrary 4x4 matrix.
678  ///
679  template <class TS = Scalar>
680  static SOPHUS_FUNC std::enable_if_t<std::is_floating_point<TS>::value, Se2>
681  fitToSe2(Eigen::Matrix3<Scalar> const& mat3x3) {
682  return Se2(
683  So2<Scalar>::fitToSo2(mat3x3.template block<2, 2>(0, 0)),
684  mat3x3.template block<2, 1>(0, 2));
685  }
686 
687  /// Returns the ith infinitesimal generators of SE(2).
688  ///
689  /// The infinitesimal generators of SE(2) are:
690  ///
691  /// ```
692  /// | 0 0 1 |
693  /// G_0 = | 0 0 0 |
694  /// | 0 0 0 |
695  ///
696  /// | 0 0 0 |
697  /// G_1 = | 0 0 1 |
698  /// | 0 0 0 |
699  ///
700  /// | 0 -1 0 |
701  /// G_2 = | 1 0 0 |
702  /// | 0 0 0 |
703  /// ```
704  ///
705  /// Precondition: ``i`` must be in 0, 1 or 2.
706  ///
707  SOPHUS_FUNC static Transformation generator(int i) {
708  SOPHUS_ASSERT(i >= 0 || i <= 2, "i should be in range [0,2].");
709  Tangent e;
710  e.setZero();
711  e[i] = Scalar(1);
712  return hat(e);
713  }
714 
715  /// hat-operator
716  ///
717  /// It takes in the 3-vector representation (= twist) and returns the
718  /// corresponding matrix representation of Lie algebra element.
719  ///
720  /// Formally, the hat()-operator of SE(3) is defined as
721  ///
722  /// ``hat(.): R^3 -> R^{3x33}, hat(a) = sum_i a_i * G_i`` (for i=0,1,2)
723  ///
724  /// with ``G_i`` being the ith infinitesimal generator of SE(2).
725  ///
726  /// The corresponding inverse is the vee()-operator, see below.
727  ///
728  SOPHUS_FUNC static Transformation hat(Tangent const& vec_a) {
729  Transformation mat_omega;
730  mat_omega.setZero();
731  mat_omega.template topLeftCorner<2, 2>() = So2<Scalar>::hat(vec_a[2]);
732  mat_omega.col(2).template head<2>() = vec_a.template head<2>();
733  return mat_omega;
734  }
735 
736  /// Lie bracket
737  ///
738  /// It computes the Lie bracket of SE(2). To be more specific, it computes
739  ///
740  /// ``[omega_1, omega_2]_se2 := vee([hat(omega_1), hat(omega_2)])``
741  ///
742  /// with ``[A,B] := AB-BA`` being the matrix commutator, ``hat(.)`` the
743  /// hat()-operator and ``vee(.)`` the vee()-operator of SE(2).
744  ///
745  SOPHUS_FUNC static Tangent lieBracket(
746  Tangent const& vec_a, Tangent const& vec_b) {
747  Eigen::Vector2<Scalar> upsilon1 = vec_a.template head<2>();
748  Eigen::Vector2<Scalar> upsilon2 = vec_b.template head<2>();
749  Scalar theta1 = vec_a[2];
750  Scalar theta2 = vec_b[2];
751 
752  return Tangent(
753  -theta1 * upsilon2[1] + theta2 * upsilon1[1],
754  theta1 * upsilon2[0] - theta2 * upsilon1[0],
755  Scalar(0));
756  }
757 
758  /// Construct pure rotation.
759  ///
760  static SOPHUS_FUNC Se2 rot(Scalar const& x) {
761  return Se2(So2<Scalar>(x), Eigen::Vector2<Scalar>::Zero());
762  }

765  ///
766  /// Translations are drawn component-wise from the range [-1, 1].
767  ///
768  template <class TUniformRandomBitGenerator>
769  static Se2 sampleUniform(TUniformRandomBitGenerator& generator) {
770  std::uniform_real_distribution<Scalar> uniform(Scalar(-1), Scalar(1));
771  return Se2(
772  So2<Scalar>::sampleUniform(generator),
773  Eigen::Vector2<Scalar>(uniform(generator), uniform(generator)));
774  }

798 
799  /// vee-operator
800  ///
801  /// It takes the 3x3-matrix representation ``Omega`` and maps it to the
802  /// corresponding 3-vector representation of Lie algebra.
803  ///
804  /// This is the inverse of the hat()-operator, see above.
805  ///
806  /// Precondition: ``Omega`` must have the following structure:
807  ///
808  /// | 0 -d a |
809  /// | d 0 b |
810  /// | 0 0 0 |
811  ///
812  SOPHUS_FUNC static Tangent vee(Transformation const& omega) {
813  SOPHUS_ASSERT(
814  omega.row(2).template lpNorm<1>() < kEpsilon<Scalar>,
815  "Omega: \n{}",
816  omega);
817  Tangent upsilon_omega;
818  upsilon_omega.template head<2>() = omega.col(2).template head<2>();
819  upsilon_omega[2] = So2<Scalar>::vee(omega.template topLeftCorner<2, 2>());
820  return upsilon_omega;
821  }
822 
823  protected:
824  SO2Member so2_; // NOLINT
825  TranslationMember translation_; // NOLINT
