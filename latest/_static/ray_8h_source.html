66  [[nodiscard]] std::optional<IntersectionResult> intersect(
67  Eigen::Hyperplane<TScalar, kN> const& plane) const {
68  using std::abs;
69  TScalar dot_prod = plane.normal().dot(this->direction_.vector());
70  if (abs(dot_prod) < sophus::kEpsilon<TScalar>) {
71  return std::nullopt;
72  }
73  IntersectionResult result;
74  result.lambda =
75  -(plane.offset() + plane.normal().dot(this->origin_)) / dot_prod;
76  result.point = this->pointAt(result.lambda);
77  return result;
78  }
79 
80  [[nodiscard]] Eigen::Matrix<TScalar, kN, 1> projection(
81  Eigen::Matrix<TScalar, kN, 1> const& point) const {
82  return origin_ +
83  direction_.getVector().dot(point - origin_) * direction_.vector();
84  }
85 
86  private:
87  Eigen::Matrix<TScalar, kN, 1> origin_;
88  UnitVector<TScalar, kN> direction_;
89 };

127 template <class TT>
128 struct ClosestApproachResult {
129  TT lambda0;
130  TT lambda1;
131  TT min_distance;
132 };
133 
134 /// For two parametric lines in lambda0 and lambda1 respectively,
135 /// ```
136 /// line_0: x(lambda0) = o0 + lambda0 * d0
137 /// line_1: y(lambda1) = o1 + lambda1 * d1
138 /// ```
139 /// returns distances [lambda0, lambda1] along the respective rays,
140 /// corresponding to the closest approach of x and y according to an l2 distance
141 /// measure. lambda0 and lambda1 may be positive or negative. If line_0 and
142 /// line_1 are parallel, returns nullopt as no unique solution exists
143 template <class TT>
144 std::optional<ClosestApproachResult<TT>> closestApproachParameters(
145  Ray3<TT> const& line_0, Ray3<TT> const& line_1) {
146  using std::abs;
147  // Closest approach when line segment connecting closest points on each line
148  // is perpendicular to both d0 and d1, thus:
149  // ```
150  // x(lambda0)-y(lambda1) = thi*(d0 X d1), for free scalar thi.
151  // => o0-o1 + lambda0*d0 - lambda1*d1 - thi*(d0Xd1) = 0
152  // => (d0|-d1|-d0Xd1).(lambda0,lambda1,thi)^T = -(o0-o1)
153  // A . x = b
154  // ```
155 
156  Eigen::Vector<TT, 3> const d0_cross_d1 =
157  line_0.direction().vector().cross(line_1.direction().vector());
158 
159  TT const d0_cross_s1_length = d0_cross_d1.norm();
160 
161  if (d0_cross_s1_length < kEpsilon<TT>) {
162  // Rays are parrallel so no unique solution exists.
163  return std::nullopt;
164  }
165 
166  Eigen::Matrix<TT, 3, 3> mat_a;
167  mat_a << line_0.direction().vector(), -line_1.direction().vector(),
168  -d0_cross_d1;
169 
170  Eigen::Vector<TT, 3> const b = line_1.origin() - line_0.origin();
171 
172  Eigen::Vector<TT, 3> const x = mat_a.lu().solve(b);
173  TT const lambda0 = x[0];
174  TT const lambda1 = x[1];
175  TT const min_distance = d0_cross_s1_length * x[2];
176 
177  return ClosestApproachResult<TT>{lambda0, lambda1, min_distance};
178 }
179 
180 /// For two lines ``line_0`` and ``line_1`` returns the mid-point of the line
181 /// segment connecting one point from each of the lines which are closest to
182 /// one another according to the l2 distance measure.
183 ///
184 /// If line_0 and line_1 are parallel, returns nullopt as no unique solution
185 /// exists
186 template <class TT>
187 std::optional<Eigen::Vector3<TT>> closestApproach(
188  Ray3<TT> const& line_0, Ray3<TT> const& line_1) {
189  auto maybe_result = closestApproachParameters(line_0, line_1);
190  if (!maybe_result) {
191  return std::nullopt;
192  }
193  return (line_0.pointAt(maybe_result->lambda0) +
194  line_1.pointAt(maybe_result->lambda1)) /
195  static_cast<TT>(2.0);
196 }
