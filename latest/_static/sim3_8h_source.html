57 /// Sim3 base type - implements Sim3 class but is storage agnostic.
58 ///
59 /// Sim(3) is the group of rotations and translation and scaling in 3d. It is
60 /// the semi-direct product of R+xSO(3) and the 3d Euclidean vector space. The
61 /// class is represented using a composition of RxSo3 for scaling plus
62 /// rotation and a 3-vector for translation.
63 ///
64 /// Sim(3) is neither compact, nor a commutative group.
65 ///
66 /// - 4x4 Eigen::Matrix representation:
67 ///
68 /// ```
69 /// | s*R t |
70 /// | o 1 |
71 /// ```
72 ///
73 /// where ``R`` is a 3x3 rotation matrix, ``s`` a positive scale factor,
74 /// ``t`` a translation 3-vector and ``o`` a 3-column vector of zeros.
75 ///
76 /// - Tangent 7-vector: [upsilon, omega, sigma],
77 ///
78 /// where ``upsilon`` is the translational velocity 3-vector and ``omega`` the
79 /// rotational velocity 3-vector, and ``sigma = log(s)``.
80 ///
81 /// - Internal 7-representation: [t0, t1, t2, qi0, qi1, qi2, qr],
82 ///
83 /// with ``t0, t1, t2`` are the translational components, and ``qi0, qi1, q2``
84 /// the imaginary vector part and ``qr1`` the real part of a non-zero
85 /// quaternion. Here the scale ``s`` is equal to the squared norm of the
86 /// quaternion ``s = |q|^2``.
87 ///
88 /// See RxSo3 for more details of the scaling + rotation representation in 3d.
89 ///
90 template <class TDerived>
91 class Sim3Base {
92  public:
93  using Scalar = typename Eigen::internal::traits<TDerived>::Scalar;
94  using TranslationType =
95  typename Eigen::internal::traits<TDerived>::TranslationType;
96  using RxSo3Type = typename Eigen::internal::traits<TDerived>::RxSo3Type;
97  using QuaternionType = typename RxSo3Type::QuaternionType;

100  /// (three for translation, three for rotation and one for scaling).

103  /// translation).

134  /// Adjoint transformation
135  ///
136  /// This function return the adjoint transformation ``Ad`` of the group
137  /// element ``A`` such that for all ``x`` it holds that
138  /// ``hat(Ad_A * x) = A * hat(x) A^{-1}``. See hat-operator below.
139  ///
140  SOPHUS_FUNC [[nodiscard]] Adjoint adj() const {
141  Eigen::Matrix3<Scalar> const r = rxso3().rotationMatrix();
142  Adjoint res;
143  res.setZero();
144  res.template block<3, 3>(0, 0) = rxso3().matrix();
145  res.template block<3, 3>(0, 3) = So3<Scalar>::hat(translation()) * r;
146  res.template block<3, 1>(0, 6) = -translation();
147 
148  res.template block<3, 3>(3, 3) = r;
149 
150  res(6, 6) = Scalar(1);
151  return res;
152  }

170  /// Logarithmic map
171  ///
172  /// Computes the logarithm, the inverse of the group exponential which maps
173  /// element of the group (rigid body transformations) to elements of the
174  /// tangent space (twist).
175  ///
176  /// To be specific, this function computes ``vee(logmat(.))`` with
177  /// ``logmat(.)`` being the matrix logarithm and ``vee(.)`` the vee-operator
178  /// of Sim(3).
179  ///
180  SOPHUS_FUNC [[nodiscard]] Tangent log() const {
181  // The derivation of the closed-form Sim(3) logarithm for is done
182  // analogously to the closed-form solution of the SE(3) logarithm, see
183  // J. Gallier, D. Xu, "Computing exponentials of skew symmetric matrices
184  // and logarithms of orthogonal matrices", IJRA 2002.
185  // https:///pdfs.semanticscholar.org/cfe3/e4b39de63c8cabd89bf3feff7f5449fc981d.pdf
186  // (Sec. 6., pp. 8)
187  Tangent res;
188  auto omega_sigma_and_theta = rxso3().logAndTheta();
189  Eigen::Vector3<Scalar> const omega =
190  omega_sigma_and_theta.tangent.template head<3>();
191  Scalar sigma = omega_sigma_and_theta.tangent[3];
192  Eigen::Matrix3<Scalar> const mat_omega = So3<Scalar>::hat(omega);
193  Eigen::Matrix3<Scalar> const w_inv = details::calcWInv<Scalar, 3>(
194  mat_omega, omega_sigma_and_theta.theta, sigma, scale());
195 
196  res.segment(0, 3) = w_inv * translation();
197  res.segment(3, 3) = omega;
198  res[6] = sigma;
199  return res;
200  }

202  /// Returns 4x4 matrix representation of the instance.
203  ///
204  /// It has the following form:
205  ///
206  /// | s*R t |
207  /// | o 1 |
208  ///
209  /// where ``R`` is a 3x3 rotation matrix, ``s`` a scale factor, ``t`` a

213  Transformation homogenious_matrix;
214  homogenious_matrix.template topLeftCorner<3, 4>() = matrix3x4();

251  /// Group action on 3-points.
252  ///
253  /// This function rotates, scales and translates a three dimensional point
254  /// ``p`` by the Sim(3) element ``(bar_sR_foo, t_bar)`` (= similarity
255  /// transformation):
256  ///
257  /// ``p_bar = bar_sR_foo * p_foo + t_bar``.
258  ///
259  template <
260  typename TPointDerived,
261  typename = typename std::enable_if<
262  IsFixedSizeVector<TPointDerived, 3>::value>::type>
263  SOPHUS_FUNC PointProduct<TPointDerived> operator*(
264  Eigen::MatrixBase<TPointDerived> const& p) const {
265  return rxso3() * p + translation();
266  }
267 
268  /// Group action on homogeneous 3-points. See above for more details.
269  ///
270  template <
271  typename THPointDerived,
272  typename = typename std::enable_if<
273  IsFixedSizeVector<THPointDerived, 4>::value>::type>
274  SOPHUS_FUNC HomogeneousPointProduct<THPointDerived> operator*(
275  Eigen::MatrixBase<THPointDerived> const& p) const {
276  PointProduct<THPointDerived> const tp =
277  rxso3() * p.template head<3>() + p(3) * translation();
278  return HomogeneousPointProduct<THPointDerived>(tp(0), tp(1), tp(2), p(3));
279  }

281  /// Group action on lines.
282  ///
283  /// This function rotates, scales and translates a parametrized line
284  /// ``l(t) = o + t * d`` by the Sim(3) element:
285  ///
286  /// Origin ``o`` is rotated, scaled and translated
287  /// Direction ``d`` is rotated
288  ///
289  SOPHUS_FUNC Line operator*(Line const& l) const {
290  Line rotated_line = rxso3() * l;
291  return Line(
292  rotated_line.origin() + translation(), rotated_line.direction());
293  }
294 
295  /// Group action on planes.
296  ///
297  /// This function rotates and translates a plane
298  /// ``n.x + d = 0`` by the Sim(3) element:
299  ///
300  /// Normal vector ``n`` is rotated
301  /// Offset ``d`` is adjusted for scale and translation
302  ///
303  SOPHUS_FUNC Hyperplane operator*(Hyperplane const& p) const {
304  Hyperplane const rotated = rxso3() * p;
305  return Hyperplane(
306  rotated.normal(),
307  rotated.offset() - translation().dot(rotated.normal()));
308  }

573  /// Returns derivative of exp(x) wrt. x.
574  ///
575  SOPHUS_FUNC static Eigen::Matrix<Scalar, kNumParameters, kDoF> dxExpX(
576  Tangent const& vec_a) {
577  Eigen::Matrix<Scalar, kNumParameters, kDoF> dx;
578 
579  static Eigen::Matrix3<Scalar> const kI = Eigen::Matrix3<Scalar>::Identity();
580  Eigen::Vector3<Scalar> const vec_omega = vec_a.template segment<3>(3);
581  Eigen::Vector3<Scalar> const vec_upsilon = vec_a.template head<3>();
582  Scalar const sigma = vec_a[6];
583  Scalar const theta = vec_omega.norm();
584 
585  Eigen::Matrix3<Scalar> const mat_omega = So3<Scalar>::hat(vec_omega);
586  Eigen::Matrix3<Scalar> const mat_omega2 = mat_omega * mat_omega;
587  Eigen::Vector3<Scalar> theta_domega;
588  if (theta < kEpsilon<Scalar>) {
589  theta_domega = Eigen::Vector3<Scalar>::Zero();
590  } else {
591  theta_domega = vec_omega / theta;
592  }
593  static Eigen::Matrix3<Scalar> const kOmegaDomega[3] = {
594  So3<Scalar>::hat(Eigen::Vector3<Scalar>::Unit(0)),
595  So3<Scalar>::hat(Eigen::Vector3<Scalar>::Unit(1)),
596  So3<Scalar>::hat(Eigen::Vector3<Scalar>::Unit(2))};
597 
598  Eigen::Matrix3<Scalar> const omega2_domega[3] = {
599  kOmegaDomega[0] * mat_omega + mat_omega * kOmegaDomega[0],
600  kOmegaDomega[1] * mat_omega + mat_omega * kOmegaDomega[1],
601  kOmegaDomega[2] * mat_omega + mat_omega * kOmegaDomega[2]};
602 
603  Eigen::Matrix3<Scalar> const w =
604  details::calcW<Scalar, 3>(mat_omega, theta, sigma);
605 
606  dx.template block<4, 3>(0, 0).setZero();

663  /// Group exponential
664  ///
665  /// This functions takes in an element of tangent space and returns the
666  /// corresponding element of the group Sim(3).
667  ///
668  /// The first three components of ``a`` represent the translational part
669  /// ``upsilon`` in the tangent space of Sim(3), the following three components
670  /// of ``a`` represents the rotation vector ``omega`` and the final component
671  /// represents the logarithm of the scaling factor ``sigma``.
672  /// To be more specific, this function computes ``expmat(hat(a))`` with
673  /// ``expmat(.)`` being the matrix exponential and ``hat(.)`` the hat-operator
674  /// of Sim(3), see below.
675  ///
676  SOPHUS_FUNC static Sim3<Scalar> exp(Tangent const& vec_a) {
677  // For the derivation of the exponential map of Sim(3) see
678  // H. Strasdat, "Local Accuracy and Global Consistency for Efficient Visual
679  // SLAM", PhD thesis, 2012.
680  // http:///hauke.strasdat.net/files/strasdat_thesis_2012.pdf (A.5, pp. 186)
681  Eigen::Vector3<Scalar> const vec_upsilon = vec_a.segment(0, 3);
682  Eigen::Vector3<Scalar> const vec_omega = vec_a.segment(3, 3);
683  Scalar const sigma = vec_a[6];
684  auto rxso3_and_theta = RxSo3<Scalar>::expAndTheta(vec_a.template tail<4>());
685  Eigen::Matrix3<Scalar> const mat_omega = So3<Scalar>::hat(vec_omega);
686 
687  Eigen::Matrix3<Scalar> const w =
688  details::calcW<Scalar, 3>(mat_omega, rxso3_and_theta.theta, sigma);
689  return Sim3<Scalar>(rxso3_and_theta.rxso3, w * vec_upsilon);
690  }

692  /// Returns the ith infinitesimal generators of Sim(3).
693  ///
694  /// The infinitesimal generators of Sim(3) are:
695  ///
696  /// ```
697  /// | 0 0 0 1 |
698  /// G_0 = | 0 0 0 0 |
699  /// | 0 0 0 0 |
700  /// | 0 0 0 0 |
701  ///
702  /// | 0 0 0 0 |
703  /// G_1 = | 0 0 0 1 |
704  /// | 0 0 0 0 |
705  /// | 0 0 0 0 |
706  ///
707  /// | 0 0 0 0 |
708  /// G_2 = | 0 0 0 0 |
709  /// | 0 0 0 1 |
710  /// | 0 0 0 0 |
711  ///
712  /// | 0 0 0 0 |
713  /// G_3 = | 0 0 -1 0 |
714  /// | 0 1 0 0 |
715  /// | 0 0 0 0 |
716  ///
717  /// | 0 0 1 0 |
718  /// G_4 = | 0 0 0 0 |
719  /// | -1 0 0 0 |
720  /// | 0 0 0 0 |
721  ///
722  /// | 0 -1 0 0 |
723  /// G_5 = | 1 0 0 0 |
724  /// | 0 0 0 0 |
725  /// | 0 0 0 0 |
726  ///
727  /// | 1 0 0 0 |
728  /// G_6 = | 0 1 0 0 |
729  /// | 0 0 1 0 |
730  /// | 0 0 0 0 |
731  /// ```
732  ///
733  /// Precondition: ``i`` must be in [0, 6].
734  ///
735  SOPHUS_FUNC static Transformation generator(int i) {
736  SOPHUS_ASSERT(i >= 0 || i <= 6, "i should be in range [0,6].");
737  Tangent e;
738  e.setZero();
739  e[i] = Scalar(1);
740  return hat(e);
741  }
742 
743  /// hat-operator
744  ///
745  /// It takes in the 7-vector representation and returns the corresponding
746  /// matrix representation of Lie algebra element.
747  ///
748  /// Formally, the hat()-operator of Sim(3) is defined as
749  ///
750  /// ``hat(.): R^7 -> R^{4x4}, hat(a) = sum_i a_i * G_i`` (for i=0,...,6)
751  ///
752  /// with ``G_i`` being the ith infinitesimal generator of Sim(3).
753  ///
754  /// The corresponding inverse is the vee()-operator, see below.
755  ///
756  SOPHUS_FUNC static Transformation hat(Tangent const& vec_a) {
757  Transformation vec_omega;
758  vec_omega.template topLeftCorner<3, 3>() =
759  RxSo3<Scalar>::hat(vec_a.template tail<4>());
760  vec_omega.col(3).template head<3>() = vec_a.template head<3>();
761  vec_omega.row(3).setZero();
762  return vec_omega;
763  }
764 
765  /// Lie bracket
766  ///
767  /// It computes the Lie bracket of Sim(3). To be more specific, it computes
768  ///
769  /// ``[omega_1, omega_2]_sim3 := vee([hat(omega_1), hat(omega_2)])``
770  ///
771  /// with ``[A,B] := AB-BA`` being the matrix commutator, ``hat(.)`` the
772  /// hat()-operator and ``vee(.)`` the vee()-operator of Sim(3).
773  ///
774  SOPHUS_FUNC static Tangent lieBracket(
775  Tangent const& vec_a, Tangent const& vec_b) {
776  Eigen::Vector3<Scalar> const vec_upsilon1 = vec_a.template head<3>();
777  Eigen::Vector3<Scalar> const vec_upsilon2 = vec_b.template head<3>();
778  Eigen::Vector3<Scalar> const vec_omega1 = vec_a.template segment<3>(3);
779  Eigen::Vector3<Scalar> const vec_omega2 = vec_b.template segment<3>(3);
780  Scalar sigma1 = vec_a[6];
781  Scalar sigma2 = vec_b[6];
782 
783  Tangent res;
784  res.template head<3>() = So3<Scalar>::hat(vec_omega1) * vec_upsilon2 +
785  So3<Scalar>::hat(vec_upsilon1) * vec_omega2 +
786  sigma1 * vec_upsilon2 - sigma2 * vec_upsilon1;
787  res.template segment<3>(3) = vec_omega1.cross(vec_omega2);
788  res[6] = Scalar(0);
789 
790  return res;
791  }
792 
793  /// Draw uniform sample from Sim(3) manifold.
794  ///
795  /// Translations are drawn component-wise from the range [-1, 1].
796  /// The scale factor is drawn uniformly in log2-space from [-1, 1],
797  /// hence the scale is in [0.5, 2].
798  ///
799  template <class TUniformRandomBitGenerator>
800  static Sim3 sampleUniform(TUniformRandomBitGenerator& generator) {
801  std::uniform_real_distribution<Scalar> uniform(Scalar(-1), Scalar(1));
802  return Sim3(
803  RxSo3<Scalar>::sampleUniform(generator),
804  Eigen::Vector3<Scalar>(
805  uniform(generator), uniform(generator), uniform(generator)));
806  }
807 
808  /// vee-operator
809  ///
810  /// It takes the 4x4-matrix representation ``Omega`` and maps it to the
811  /// corresponding 7-vector representation of Lie algebra.
812  ///
813  /// This is the inverse of the hat()-operator, see above.
814  ///
815  /// Precondition: ``Omega`` must have the following structure:
816  ///
817  /// | g -f e a |
818  /// | f g -d b |
819  /// | -e d g c |
820  /// | 0 0 0 0 |
821  ///
822  SOPHUS_FUNC static Tangent vee(Transformation const& mat_omega) {
823  Tangent vec_upsilon_omega_sigma;
824  vec_upsilon_omega_sigma.template head<3>() =
825  mat_omega.col(3).template head<3>();
826  vec_upsilon_omega_sigma.template tail<4>() =
827  RxSo3<Scalar>::vee(mat_omega.template topLeftCorner<3, 3>());
828  return vec_upsilon_omega_sigma;
829  }
830 
831  protected:
832  RxSo3Member rxso3_; // NOLINT
833  TranslationMember translation_; // NOLINT
834 };
