10 /// ImageView and MutImageView, non owning images types.
11 ///
12 /// See image.h for Image and MutImage, owning images types.
13 ///
14 /// Note that it is a conscious API decision to follow "shallow-compare" type
15 /// semantic for ImageView, MutImageView, Image and MutImage. Similar
16 /// "shallow-compare" types are: std::span (shallow-compare reference type), and
17 /// std::unique_ptr (shallow compare unique ownership type).
18 /// This is in contrast to regular types such as std::vector, std::string and
19 /// reference types which mimic regular type semantic such as std::string_view.
20 /// Also see https://abseil.io/blog/20180531-regular-types.

40 /// View of a mutable image, which does not own the data.
41 ///
42 /// The API of MutImageView allows for read and write access.
43 ///
44 /// MutImageView is nullable. In that case `this->isEmpty()` is true.
45 ///
46 ///
47 /// Details on equality comparison, the state of the object, and
48 /// const-correctness.
49 ///
50 /// MutImageView is a "shallow-compare type" similar to std::span<Pixel> and
51 /// std::unique_ptr<Pixel>. As ImageView, its state consists of the image shape
52 /// as well as the pointer address, and comparing those entities establishes
53 /// equality comparisons. Furthermore, giving mutable access to pixels is
54 /// considered a const operation, as in
55 ///
56 /// ```TPixel& mut(int u, int v) const```
57 ///
58 /// since this merely allows for changing a pixel value, but not its state
59 /// (data location and layout).
60 template <class TPixel>
61 class MutImageView : public ImageView<TPixel> {
62  public:
63  /// Default constructor creates an empty image.
64  MutImageView() = default;
65 
66  /// Creates view from shape and pointer to first pixel.
67  MutImageView(ImageShape shape, TPixel* ptr) noexcept
68  : ImageView<TPixel>(shape, ptr) {}
69 
70  /// Creates view from image size and pointer to first pixel. The image is
71  /// assumed to be contiguous and the pitch is set accordingly.
72  MutImageView(sophus::ImageSize image_size, TPixel* ptr) noexcept
73  : ImageView<TPixel>(image_size, ptr) {}
74 
75  /// Creates mutable view from view.
76  ///
77  /// It is the user's responsibility to make sure that the data owned by
78  /// the view can be modified safely.
79  [[nodiscard]] static MutImageView unsafeConstCast(ImageView<TPixel> view) {
80  return MutImageView(view.shape(), const_cast<TPixel*>(view.ptr()));
81  }
82 
83  /// Returns ImageView(*this).
84  ///
85  /// Returns non-mut version of view.
86  [[nodiscard]] ImageView<TPixel> view() const {
87  return ImageView<TPixel>(this->shape(), this->ptr());
88  }
89 
90  /// Copies data from view into this.
91  ///
92  /// Preconditions:
93  /// * this->isEmpty() == view.isEmpty()
94  /// * this->size() == view.size()
95  ///
96  /// No-op if view is empty.
97  void copyDataFrom(ImageView<TPixel> view) const {
98  SOPHUS_ASSERT_EQ(this->isEmpty(), view.isEmpty());
99 
100  if (this->isEmpty()) {
101  return;
102  }
103  SOPHUS_ASSERT_EQ(this->imageSize(), view.imageSize());
104  details::pitchedCopy(
105  (uint8_t*)this->ptr(),
106  this->shape().pitchBytes(),
107  (uint8_t const*)view.ptr(),
108  view.shape().pitchBytes(),
109  this->imageSize(),
110  sizeof(TPixel));
111  }
112 
113  /// Returns v-th row pointer of mutable pixel.
114  [[nodiscard]] TPixel* rowPtrMut(int v) const {
115  return (TPixel*)((uint8_t*)(this->ptr()) + v * this->shape_.pitchBytes());
116  }
117 
118  /// Mutable accessor to pixel u, v.
119  ///
120  /// Precondition: u must be in [0, width) and v must be in [0, height).
121  /// Silent UB on failure.
122  [[nodiscard]] TPixel& mut(int u, int v) const { return rowPtrMut(v)[u]; }
123 
124  [[nodiscard]] TPixel& mut(Eigen::Vector2i uv) const {
125  return mut(uv[0], uv[1]);
126  }
127 
128  /// Mutates each pixel of this with given unary operation
129  ///
130  /// Preconditions: this must not be empty.
131  template <class TUnaryOperation>
132  void mutate(TUnaryOperation const& unary_op) const {
133  SOPHUS_ASSERT(!this->isEmpty());
134 
135  for (int v = 0; v < this->shape_.height(); ++v) {
136  TPixel* p = this->rowPtrMut(v);
137  TPixel const* end_of_row = p + this->shape_.width();
138  for (; p != end_of_row; ++p) {
139  *p = unary_op(*p);
140  }
141  }
142  }
143 
144  /// For each pixel in `this` with coordinates (u,v), populates with the user
145  /// provided function, evaluated as `uv_op(u,v)`, where u and v are integers
146  /// such that u in [0, width), v in [0, height)
147  ///
148  /// Preconditions: this must not be empty.
149  template <class TUVOperation>
150  void generate(TUVOperation const& uv_op) const {
151  SOPHUS_ASSERT(!this->isEmpty());
152 
153  for (int v = 0; v < this->shape_.height(); ++v) {
154  TPixel* p = this->rowPtrMut(v);
155  TPixel const* end_of_row = p + this->shape_.width();
156  for (int u = 0; p != end_of_row; ++p, ++u) {
157  *p = uv_op(u, v);
158  }
159  }
160  }
161 
162  /// Transforms view using unary operation and assigns result to this.
163  ///
164  /// Preconditions:
165  /// - this must not be empty.
166  /// - this->imageSize() == view.imageSize()
167  template <class TOtherPixel, class TUnaryOperation>
168  void transformFrom(
169  ImageView<TOtherPixel> view, TUnaryOperation const& unary_op) const {
170  SOPHUS_ASSERT(!this->isEmpty());
171  SOPHUS_ASSERT_EQ(view.imageSize(), this->imageSize());
172 
173  for (int v = 0; v < this->shape_.height(); ++v) {
174  TPixel* mut_p = this->rowPtrMut(v);
175  TOtherPixel const* p = view.rowPtr(v);
176  TOtherPixel const* end_of_row = p + view.shape().width();
177 
178  for (; p != end_of_row; ++p, ++mut_p) {
179  *mut_p = unary_op(*p);
180  }
181  }
182  }
183 
184  /// Transforms two views using binary operation and assigns result to this.
185  ///
186  /// Preconditions:
187  /// - this must not be empty.
188  /// - this->imageSize() == lhs.imageSize() == rhs.imageSize()
189  template <class TLhsPixel, class TRhsPixel, class TBinaryOperation>
190  void transformFrom(
191  ImageView<TLhsPixel> lhs,
192  ImageView<TRhsPixel> rhs,
193  TBinaryOperation const& binary_op) const {
194  SOPHUS_ASSERT(!this->isEmpty());
195  SOPHUS_ASSERT_EQ(lhs.imageSize(), this->imageSize());
196  SOPHUS_ASSERT_EQ(rhs.imageSize(), this->imageSize());
197 
198  for (int v = 0; v < this->shape_.height(); ++v) {
199  TPixel* mut_p = this->rowPtrMut(v);
200  TLhsPixel const* lhs_p = lhs.rowPtr(v);
201  TRhsPixel const* rhs_p = rhs.rowPtr(v);
202 
203  for (int u = 0; u < this->shape_.width(); ++u) {
204  mut_p[u] = binary_op(lhs_p[u], rhs_p[u]);
205  }
206  }
207  }
208 
209  /// Populates every pixel of this with val.
210  ///
211  /// Preconditions: this must not be empty.
212  void fill(TPixel const& val) const {
213  mutate([&](TPixel const& /*unused*/) { return val; });
214  }
215 
216  /// Returns pointer of mutable data to first pixel.
217  [[nodiscard]] TPixel* ptrMut() const {
218  return const_cast<TPixel*>(ImageView<TPixel>::ptr());
219  }
220 
221  /// Returns mutable subview.
222  [[nodiscard]] MutImageView mutSubview(
223  Eigen::Vector2i uv, sophus::ImageSize size) const {
224  SOPHUS_ASSERT(this->colInBounds(uv[0]));
225  SOPHUS_ASSERT(this->rowInBounds(uv[1]));
226  SOPHUS_ASSERT_LE(uv.x() + size.width, this->shape().width());
227  SOPHUS_ASSERT_LE(uv.y() + size.height, this->shape().height());
228 
229  return MutImageView(
230  ImageShape::makeFromSizeAndPitch<TPixel>(
231  size, this->shape().pitchBytes()),
232  this->rowPtrMut(uv.y()) + uv.x());
233  }
234 };

238 template <class TPixel>
239 auto checkedPixelAccessMut(
240  MutImageView<TPixel> const& view,
241  int u,
242  int v,
243  std::string const& file,
244  int line,
245  std::string const& str) -> TPixel& {
246  if (!view.colInBounds(u) || !view.rowInBounds(v)) {
247  FARM_IMPL_LOG_PRINTLN("[SOPHUS_PIXEL_MUT in {}:{}]", file, line);
248  FARM_IMPL_LOG_PRINTLN(
249  "pixel `{},{}` not in image with size {} x {}",
250  u,
251  v,
252  view.imageSize().width,
253  view.imageSize().height);
254  if (!str.empty()) {
255  ::fmt::print(stderr, "{}", str);
256  }
257  FARM_IMPL_ABORT();
258  }
259  return view.mut(u, v);
260 }
