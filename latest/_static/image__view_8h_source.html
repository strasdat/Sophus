10 /// ImageView and MutImageView, non owning images types.
11 ///
12 /// See image.h for Image and MutImage, owning images types.
13 ///
14 /// Note that it is a conscious API decision to follow "shallow-compare" type
15 /// semantic for ImageView, MutImageView, Image and MutImage. Similar
16 /// "shallow-compare" types are: std::span (shallow-compare reference type), and
17 /// std::unique_ptr (shallow compare unique ownership type).
18 /// This is in contrast to regular types such as std::vector, std::string and
19 /// reference types which mimic regular type semantic such as std::string_view.

37 /// A view of an (immutable) image, which does not own the data.
38 ///
39 /// The API of ImageView allows for read-only access. There is an escape hook
40 /// for write access, see MutImageView::unsafeConstCast below.
41 ///
42 /// ImageViews are nullable. In that case `this->isEmpty()` is true.
43 ///
44 ///
45 /// Details on equality comparison, the state of the object, and
46 /// const-correctness.
47 ///
48 /// ImageView is a "shallow-compare type" similar to std::span<Pixel const> and
49 /// std::unique_ptr<Pixel const>. In particular, we define that the state of an
50 /// ImageView instance consists of the layout of the image ``layout_`` (see
51 /// ImageLayout) and the pointer address to the first pixel ``ptr_``. No
52 /// public member method can change the pointer nor the layout, hence they are
53 /// all marked const.
54 template <class TPixel>
55 struct ImageView {
56  using Pixel = TPixel;
57 
58  /// Default constructor creates an empty image.
59  ImageView() = default;
60 
61  /// Creates view from layout and pointer to first pixel.
62  ImageView(ImageLayout layout, TPixel const* ptr) noexcept
63  : layout_(layout), ptr_(ptr) {}
64 
65  /// Creates view from image size and pointer to first pixel. The image is
66  /// assumed to be contiguous and the pitch is set accordingly.
67  ImageView(sophus::ImageSize image_size, TPixel const* ptr) noexcept
68  : ImageView(ImageLayout::makeFromSize<TPixel>(image_size), ptr) {}
69 
70  /// Returns true if view is empty.
71  [[nodiscard]] bool isEmpty() const { return this->ptr_ == nullptr; }
72 
73  /// Returns true if view is contiguous.
74  [[nodiscard]] bool isContiguous() const {
75  return imageSize().width * sizeof(TPixel) == layout().pitchBytes();
76  }
77 
78  /// Returns ImageSize.
79  /// It is {0,0} if view is empty.
80  [[nodiscard]] sophus::ImageSize const& imageSize() const {
81  return layout_.imageSize();
82  }
83 
84  /// Returns ImageLayout.
85  /// It is {{0,0}, 0} is view is empty.
86  [[nodiscard]] ImageLayout const& layout() const { return layout_; }
87 
88  [[nodiscard]] int width() const { return layout().width(); }
89  [[nodiscard]] int height() const { return layout().height(); }
90  [[nodiscard]] size_t pitchBytes() const { return layout().pitchBytes(); }
91 
92  /// Returns true if u is in [0, width).
93  [[nodiscard]] bool colInBounds(int u) const {
94  return u >= 0 && u < layout_.width();
95  }
96 
97  /// Returns true if v is in [0, height).
98  [[nodiscard]] bool rowInBounds(int v) const {
99  return v >= 0 && v < layout_.height();
100  }
101 
102  /// Returns v-th row pointer.
103  ///
104  /// Precondition: v must be in [0, height).
105  [[nodiscard]] TPixel const* rowPtr(int v) const {
106  return (TPixel*)((uint8_t*)(ptr_) + v * layout_.pitchBytes());
107  }
108 
109  /// Returns pixel u, v.
110  ///
111  /// Precondition: u must be in [0, width) and v must be in [0, height).
112  ///
113  /// Note:
114  /// * No panic if u.v is invalid,
115  ///
116  /// This is not the most necessarily the efficient function to call - e.g.
117  /// when iterating over the whole image. Use the following instead:
118  ///
119  /// for (int v=0; v<view.layout().height(); ++v) {
120  /// TPixel const* row = img.rowPtr(v);
121  /// for (int u=0; u<view.layout().width(); ++u) {
122  /// PixetT p = row[u];
123  /// }
124  /// }
125  [[nodiscard]] TPixel const& operator()(int u, int v) const {
126  return rowPtr(v)[u];
127  }
128 
129  [[nodiscard]] TPixel const& operator()(Eigen::Vector2i uv) const {
130  return this->operator()(uv[0], uv[1]);
131  }
132 
133  /// Returns pointer to first pixel.
134  [[nodiscard]] TPixel const* ptr() const { return ptr_; }
135 
136  /// Returns subview.
137  [[nodiscard]] ImageView subview(
138  Eigen::Vector2i uv, sophus::ImageSize size) const {
139  SOPHUS_ASSERT(colInBounds(uv[0]));
140  SOPHUS_ASSERT(rowInBounds(uv[1]));
141  SOPHUS_ASSERT_LE(uv.x() + size.width, layout_.width());
142  SOPHUS_ASSERT_LE(uv.y() + size.height, layout_.height());
143  return ImageView(
144  ImageLayout::makeFromSizeAndPitch<TPixel>(size, layout_.pitchBytes()),
145  rowPtr(uv.y()) + uv.x());
146  }

148  /// Performs reduction / fold on image view.
149  template <class TFunc>
150  void visit(TFunc const& user_function) const {
151  SOPHUS_ASSERT(!this->isEmpty());
152 
153  for (int v = 0; v < this->layout_.height(); ++v) {
154  TPixel const* p = this->rowPtr(v);
155  TPixel const* end_of_row = p + this->layout_.width();
156  for (; p != end_of_row; ++p) {
157  user_function(*p);
158  }
159  }
160  }
161 
162  /// Performs reduction / fold on image view.
163  template <class TReduceOp, class TVal>
164  [[nodiscard]] TVal reduce(
165  TReduceOp const& reduce_op, TVal val = TVal{}) const {
166  SOPHUS_ASSERT(!this->isEmpty()); // NOLINT
167 
168  for (int v = 0; v < this->layout_.height(); ++v) {
169  TPixel const* p = this->rowPtr(v);
170  TPixel const* end_of_row = p + this->layout_.width();
171  for (; p != end_of_row; ++p) {
172  reduce_op(*p, val);
173  }
174  }
175  return val;
176  }
177 
178  /// Performs reduction / fold on image view with short circuit condition.
179  template <class TShortCircuitReduceOp, class TVal>
180  [[nodiscard]] TVal shortCircuitReduce(
181  TShortCircuitReduceOp const& short_circuit_reduce_op,
182  TVal val = TVal{}) const {
183  SOPHUS_ASSERT(!this->isEmpty());
184 
185  for (int v = 0; v < this->layout_.height(); ++v) {
186  TPixel const* p = this->rowPtr(v);
187  TPixel const* end_of_row = p + this->layout_.width();
188  for (; p != end_of_row; ++p) {
189  if (short_circuit_reduce_op(*p, val)) {
190  return val;
191  }
192  }
193  }
194  return val;
195  }
196 
197  /// The equality operator is deleted to avoid confusion. Since ImageView is a
198  /// "shallow-copy" type, a consistently defined equality would check for
199  /// equality of its (shallow) state:
200  ///
201  /// ```this->layout_ == rhs.layout() && this->ptr_ == rhs.ptr_````
202  ///
203  /// However, some users might expect that equality would check for pixel
204  /// values equality and return true for identical copies of data blocks.
205  ///
206  /// Here we follow std::span which also does not offer equality comparions.
207  bool operator==(ImageView const& rhs) const = delete;
208 
209  /// The in-equality operator is deleted to avoid confusion.
210  bool operator!=(ImageView const& rhs) const = delete;
211 
212  /// Returns true both views have the same size and contain the same data.
213  [[nodiscard]] bool hasSameData(ImageView const& rhs) const {
214  if (!(this->imageSize() == rhs.imageSize())) {
215  return false;
216  }
217  for (int v = 0; v < this->layout_.height(); ++v) {
218  TPixel const* p = this->rowPtr(v);
219  TPixel const* rhs_p = rhs.rowPtr(v);
220 
221  TPixel const* end_of_row = p + this->layout_.width();
222  for (; p != end_of_row; ++p, ++rhs_p) {
223  if (*p != *rhs_p) {
224  return false;
225  }
226  }
227  }
228  return true;
229  }
230 
231  protected:
232  /// Resets view such that it is empty.
233  void setViewToEmpty() { *this = {}; }
234 
235  ImageLayout layout_ = {}; // NOLINT
236  TPixel const* ptr_ = nullptr; // NOLINT
237 
238  private:
239  template <class TT, class TAllocator>
240  friend class MutImage;
241 
242  template <class TT, class TAllocator>
243  friend class Image;
244 };
