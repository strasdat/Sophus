39  template <class TParamsTypeT, class TPointTypeT>
40  static PixelImage<typename TPointTypeT::Scalar> distort(
41  Eigen::MatrixBase<TParamsTypeT> const& params,
42  Eigen::MatrixBase<TPointTypeT> const& proj_point_in_camera_z1_plane) {
43  static_assert(
44  TParamsTypeT::ColsAtCompileTime == 1, "params must be a column-vector");
45  static_assert(
46  TParamsTypeT::RowsAtCompileTime == kNumParams,
47  "params must have exactly kNumParams rows");
48  static_assert(
49  TPointTypeT::ColsAtCompileTime == 1,
50  "point_camera must be a column-vector");
51  static_assert(
52  TPointTypeT::RowsAtCompileTime == 2,
53  "point_camera must have exactly 2 columns");
54 
55  auto const k0 = params[4];
56  auto const k1 = params[5];
57  auto const k2 = params[6];
58  auto const k3 = params[7];
59 
60  auto const radius_squared =
61  proj_point_in_camera_z1_plane[0] * proj_point_in_camera_z1_plane[0] +
62  proj_point_in_camera_z1_plane[1] * proj_point_in_camera_z1_plane[1];
63  using std::atan2;
64  using std::sqrt;
65 
66  if (radius_squared > sophus::kEpsilonF64) {
67  auto const radius = sqrt(radius_squared);
68  auto const radius_inverse = 1.0 / radius;
69  auto const theta = atan2(radius, typename TPointTypeT::Scalar(1.0));
70  auto const theta2 = theta * theta;
71  auto const theta4 = theta2 * theta2;
72  auto const theta6 = theta4 * theta2;
73  auto const theta8 = theta4 * theta4;
74  auto const r_distorted =
75  theta * (1.0 + k0 * theta2 + k1 * theta4 + k2 * theta6 + k3 * theta8);
76  auto const scaling = r_distorted * radius_inverse;
77 
78  return scaling * proj_point_in_camera_z1_plane.cwiseProduct(
79  params.template head<2>()) +
80  params.template segment<2>(2);
81  } // linearize r around radius=0
82 
83  return AffineTransform::distort(
84 
85  params.template head<4>(), proj_point_in_camera_z1_plane);
86  }

88  template <class TScalar>
89  static ProjInCameraZ1Plane<TScalar> undistort(
90  Params<TScalar> const& params, PixelImage<TScalar> const& pixel_image) {
91  using std::abs;
92  using std::sqrt;
93  using std::tan;
94 
95  // Undistortion
96  const TScalar fu = params[0];
97  const TScalar fv = params[1];
98  const TScalar u0 = params[2];
99  const TScalar v0 = params[3];
100 
101  const TScalar k0 = params[4];
102  const TScalar k1 = params[5];
103  const TScalar k2 = params[6];
104  const TScalar k3 = params[7];
105 
106  const TScalar un = (pixel_image(0) - u0) / fu;
107  const TScalar vn = (pixel_image(1) - v0) / fv;
108  const TScalar rth2 = un * un + vn * vn;
109 
110  if (rth2 < sophus::kEpsilon<TScalar> * sophus::kEpsilon<TScalar>) {
111  return Eigen::Matrix<TScalar, 2, 1>(un, vn);
112  }
113 
114  const TScalar rth = sqrt(rth2);
115 
116  // Use Newtons method to solve for theta, 50 iterations max
117  TScalar th = sqrt(rth);
118  for (int i = 0; i < 500; ++i) {
119  const TScalar th2 = th * th;
120  const TScalar th4 = th2 * th2;
121  const TScalar th6 = th4 * th2;
122  const TScalar th8 = th4 * th4;
123 
124  const TScalar thd =
125  th * (TScalar(1.0) + k0 * th2 + k1 * th4 + k2 * th6 + k3 * th8);
126 
127  const TScalar d_thd_wtr_th =
128  TScalar(1.0) + TScalar(3.0) * k0 * th2 + TScalar(5.0) * k1 * th4 +
129  TScalar(7.0) * k2 * th6 + TScalar(9.0) * k3 * th8;
130 
131  const TScalar step = (thd - rth) / d_thd_wtr_th;
132  th -= step;
133  // has converged?
134  if (abs(jet_helpers::GetValue<TScalar>::impl(step)) <
135  sophus::kEpsilon<TScalar>) {
136  break;
137  }
138  }
139 
140  TScalar radius_undistorted = tan(th);
141 
142  if (radius_undistorted < TScalar(0.0)) {
143  return Eigen::Matrix<TScalar, 2, 1>(
144  -radius_undistorted * un / rth, -radius_undistorted * vn / rth);
145  }
146  return Eigen::Matrix<TScalar, 2, 1>(
147  radius_undistorted * un / rth, radius_undistorted * vn / rth);
148  }

150  template <class TParamsTypeT, class TPointTypeT>
151  static Eigen::Matrix<typename TPointTypeT::Scalar, 2, 2> dxDistort(
152  Eigen::MatrixBase<TParamsTypeT> const& params,
153  Eigen::MatrixBase<TPointTypeT> const& proj_point_in_camera_z1_plane) {
154  static_assert(
155  TParamsTypeT::ColsAtCompileTime == 1, "params must be a column-vector");
156  static_assert(
157  TParamsTypeT::RowsAtCompileTime == kNumParams,
158  "params must have exactly kNumParams rows");
159  static_assert(
160  TPointTypeT::ColsAtCompileTime == 1,
161  "point_camera must be a column-vector");
162  static_assert(
163  TPointTypeT::RowsAtCompileTime == 2,
164  "point_camera must have exactly 2 columns");
165  using Scalar = typename TPointTypeT::Scalar;
166 
167  Scalar a = proj_point_in_camera_z1_plane[0];
168  Scalar b = proj_point_in_camera_z1_plane[1];
169  Scalar const fx = params[0];
170  Scalar const fy = params[1];
171  Eigen::Matrix<Scalar, kNumDistortionParams, 1> k =
172  params.template tail<kNumDistortionParams>();
173 
174  auto const radius_squared = a * a + b * b;
175  using std::atan2;
176  using std::sqrt;
177 
178  Eigen::Matrix<typename TPointTypeT::Scalar, 2, 2> dx;
179 
180  if (radius_squared < sophus::kEpsilonSqrtF64) {
181  // clang-format off
182  dx << //
183  fx, 0,
184  0, fy;
185  // clang-format on
186  return dx;
187  }
188 
189  using std::atan2;
190  using std::pow;
191  using std::sqrt;
192 
193  Scalar const c0 = pow(a, 2);
194  Scalar const c1 = pow(b, 2);
195  Scalar const c2 = c0 + c1;
196  Scalar const c3 = pow(c2, 5.0 / 2.0);
197  Scalar const c4 = c2 + 1;
198  Scalar const c5 = atan(sqrt(c2));
199  Scalar const c6 = pow(c5, 2);
200  Scalar const c7 = c6 * k[0];
201  Scalar const c8 = pow(c5, 4);
202  Scalar const c9 = c8 * k[1];
203  Scalar const c10 = pow(c5, 6);
204  Scalar const c11 = c10 * k[2];
205  Scalar const c12 = pow(c5, 8) * k[3];
206  Scalar const c13 = 1.0 * c4 * c5 * (c11 + c12 + c7 + c9 + 1.0);
207  Scalar const c14 = c13 * c3;
208  Scalar const c15 = pow(c2, 3.0 / 2.0);
209  Scalar const c16 = c13 * c15;
210  Scalar const c17 =
211  1.0 * c11 + 1.0 * c12 +
212  2.0 * c6 * (4 * c10 * k[3] + 2 * c6 * k[1] + 3 * c8 * k[2] + k[0]) +
213  1.0 * c7 + 1.0 * c9 + 1.0;
214  Scalar const c18 = c17 * pow(c2, 2);
215  Scalar const c19 = 1.0 / c4;
216  Scalar const c20 = c19 / pow(c2, 3);
217  Scalar const c21 = a * b * c19 * (-c13 * c2 + c15 * c17) / c3;
218 
219  dx(0, 0) = c20 * fx * (-c0 * c16 + c0 * c18 + c14);
220  dx(0, 1) = c21 * fx;
221 
222  dx(1, 0) = c21 * fy;
223  dx(1, 1) = c20 * fy * (-c1 * c16 + c1 * c18 + c14);
224 
225  return dx;
226  }
227 };
