26 /// Image read-only access to pixels and shared ownership, hence cheap to copy.
27 /// Type is nullable.
28 ///
29 /// Image has close interop with DynImage (see below).
30 template <class TPixel, class TAllocator = Eigen::aligned_allocator<uint8_t>>
31 class Image : public ImageView<TPixel> {
32  public:
33  /// Constructs empty image.
34  Image() = default;
35 
36  /// Moves MutImage into this.
37  /// By design not "explicit".
38  Image(MutImage<TPixel, TAllocator>&& image) noexcept
39  : ImageView<TPixel>(image.view()) {
40  if (!image.isEmpty()) {
41  this->shared_ = std::move(image.unique_);
42  this->ptr_ = reinterpret_cast<TPixel*>(this->shared_.get());
43  image.setViewToEmpty();
44  }
45  }
46 
47  /// Creates contiguous copy from view.
48  ///
49  /// If view is not empty, memory allocation will happen.
50  [[nodiscard]] static Image makeCopyFrom(ImageView<TPixel> const& view) {
51  return Image(MutImage<TPixel>::makeCopyFrom(view));
52  }
53 
54  /// Allocated and generates image from provided function taking u,v indices
55  ///
56  /// Memory allocation will happen.
57  template <class TUVOperation>
58  [[nodiscard]] static Image makeGenerative(
59  ImageSize size, TUVOperation const& uv_op) {
60  MutImage<TPixel> mut_image(size);
61  mut_image.generate(uv_op);
62  return mut_image;
63  }

65  /// Creates new Image given view and unary transform function.

88  ///
