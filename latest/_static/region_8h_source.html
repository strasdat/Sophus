43 /// A region is a closed interval [a, b] with a being the lower bound (=min) and
44 /// b being the upper bound (=max).
45 ///
46 /// Here, the bounds a, b sre either boths scalars (e.g. floats, doubles)
47 /// or fixed length vectors/arrays (such as Eigen::Vector3f or Eigen::Array2d).
48 ///
49 /// Special case for integer numbers:
50 ///
51 /// An integer number X is considered being equivalent to a real region
52 /// [X-0.5, X+0.5].
53 template <PointType TPoint>
54 class Region {
55  public:
56  static bool constexpr kIsInteger = PointTraits<TPoint>::kIsInteger;
57  using Point = TPoint;
58  using Scalar = typename PointTraits<TPoint>::Scalar;
59 
60  static int constexpr kRows = PointTraits<TPoint>::kRows;
61  static int constexpr kCols = PointTraits<TPoint>::kCols;
62  static_assert(kRows >= 1);
63  static_assert(kCols >= 1);
64  static int constexpr kDim = kRows * kCols;
65 
66  /// Creates and empty region.
67  static Region<TPoint> empty() noexcept {
68  Region region(UninitTag{});
69  region.min_ = std::nullopt;
70  region.max_ = std::nullopt;
71  return region;
72  }

74  /// Creates unbounded region.
75  ///
76  /// If TPoint is floating point, the region is [-inf, +inf].
77  static Region<TPoint> unbounded() noexcept {
78  Region region(UninitTag{});
79  region.min_ = PointTraits<TPoint>::min();
80  region.max_ = PointTraits<TPoint>::max();

84  /// Creates a region from a given point.
85  ///
86  /// If TPoint is a floating point then the region is considered degenerated.
87  static Region<TPoint> from(TPoint const& p) noexcept {
88  auto region = Region<TPoint>::empty();
89  SOPHUS_ASSERT(!isNan(p));
90  region.extend(p);
91  return region;
92  }

94  /// Creates Region from two points, min and max.
95  ///
96  /// The points min, max need not to be ordered. After construction it will be
97  /// true that this->min() <= this->max().
98  static Region<TPoint> fromMinMax(
99  TPoint const& min, TPoint const& max) noexcept {
100  auto region = Region<TPoint>::empty();
101  SOPHUS_ASSERT(!isNan(min));
102  SOPHUS_ASSERT(!isNan(max));
103  region.extend(min);
104  region.extend(max);
105  return region;
106  }
107 
108  /// Convenient constructor to create a Segment from two points p1 and p2.
109  ///
110  /// The points need not to be ordered. After construction it will be true
111  /// that this->min() <= this->max().
112  ///
113  /// Note: This constructor is only available for scalar regions. For
114  /// multi-dim regions, use the fromMinMax() factory instead.,
115  template <class TScalar>
116  requires std::is_same_v<TPoint, TScalar> Region(
117  TScalar const& p1, TScalar const& p2)
118  noexcept : min_{p1}, max_{p2} {
119  SOPHUS_ASSERT(!isNan(p1));
120  SOPHUS_ASSERT(!isNan(p2));
121  this->extend(p2);
122  }

209  ///

219  /// Returns true if the region contains the given point.
220  [[nodiscard]] bool contains(TPoint const& point) const noexcept {
221  if (this->isEmpty()) {
222  return false;
223  }
224  return isLessEqual(min(), point) && isLessEqual(point, max());
225  }
226 
227  /// Returns the range of the region.
228  ///
229  /// It is zero if the region is not proper.
230  [[nodiscard]] TPoint range() const noexcept {
231  if (this->isEmpty()) {
232  return zero<TPoint>();
233  }
234  if constexpr (kIsInteger) {
235  // For integers, we consider e.g. {2} == [1.5, 2.5] hence range of
236  // Region(2, 3) == [1.5, 3.5] is 2.
237  return plus(eval(this->max() - this->min()), 1);
238  }
239  return eval(max() - min());
240  }
241 
242  /// Returns the mid point.
243  ///
244  /// Note: If TPoint is an integer point then the result will be rounded to
245  /// the closed integer.
246  [[nodiscard]] TPoint mid() const noexcept {
247  return eval(min() + range() / 2);
248  }

250  /// Extends this by other region.
251  Region<TPoint> extend(Region const& other) noexcept {
252  if (this->isEmpty()) {
253  return other;
254  }
255  if (other.isEmpty()) {
256  return *this;
257  }
258  return fromMinMax(
259  sophus::min(min(), other.min()), sophus::max(max(), other.max()));
260  }
261 
262  /// Extends this by given point.
263  Region<TPoint>& extend(TPoint const& point) noexcept {
264  if (this->isEmpty()) {
265  *this = from(point);
266  } else {
267  *this = fromMinMax(sophus::min(min(), point), sophus::max(max(), point));
268  }
269  return *this;
270  }
271 
272  /// Returns translated region.
273  [[nodiscard]] Region<TPoint> translated(TPoint const& p) const noexcept {
274  if (this->isEmpty()) {
275  return *this;
276  }
277  return Region<TPoint>::fromMinMax(this->min() + p, this->max() + p);
278  }

280  template <class TOtherPoint>
281  Region<TOtherPoint> cast() const noexcept {
282  if (isEmpty()) {
283  return Region<TOtherPoint>::empty();
284  }
285  if constexpr (kIsInteger == Region<TOtherPoint>::kIsInteger) {
286  // case 1: floating => floating and integer => integer is trivial
287  return Region<TOtherPoint>(
288  sophus::cast<TOtherPoint>(min()), sophus::cast<TOtherPoint>(max()));
289  }
290  if constexpr (kIsInteger && !Region<TOtherPoint>::kIsInteger) {
291  // case 2: integer to floating.
292  //
293  // example: [2, 5] -> [1.5, 5.5]
294  return Region<TOtherPoint>(
295  plus(sophus::cast<TOtherPoint>(min()), -0.5),
296  plus(sophus::cast<TOtherPoint>(max()), 0.5));
297  }
298  // case 3: floating to integer.
299  static_assert(
300  kIsInteger || !Region<TOtherPoint>::kIsInteger,
301  "For floating to integer: call encloseCast() or roundCast() "
302  "instead.");
303  }

305  /// Returns the smallest integer region which contains this.
306  ///
307  /// example: [1.2, 1.3] -> [1, 2]
308  template <class TOtherPoint>
309  Region<TOtherPoint> encloseCast() const noexcept {
310  static_assert(!kIsInteger && Region<TOtherPoint>::kIsInteger);
311  if (this->isEmpty()) {
312  return Region<TOtherPoint>::empty();
313  }
314  return Region<TOtherPoint>(
315  sophus::cast<TOtherPoint>(sophus::floor(min())),
316  sophus::cast<TOtherPoint>(sophus::ceil(max())));
317  }
318 
319  /// Rounds given region bounds and returns resulting integer region.
320  ///
321  /// example: [1.2, 2.3] -> [1, 2]
322  /// example: [1.1, 2.7] -> [1, 3]
323  template <class TOtherPoint>
324  Region<TOtherPoint> roundCast() const noexcept {
325  static_assert(!kIsInteger && Region<TOtherPoint>::kIsInteger);
326  if (this->isEmpty()) {
327  return Region<TOtherPoint>::empty();
328  }
329  return Region<TOtherPoint>(
330  sophus::cast<TOtherPoint>(sophus::round(min())),
331  sophus::cast<TOtherPoint>(sophus::round(max())));
332  }

334  /// Returns true if region is empty.
335  [[nodiscard]] bool isEmpty() const noexcept { return !bool(min_); }
336 
337  /// Returns true if region contains a single floating point number.
338  [[nodiscard]] bool isDegenerated() const noexcept {
339  if (this->isEmpty()) {
340  return false;
341  }
342  return !kIsInteger && allTrue(min() == max());
343  }
344 
345  /// Returns true if region is neither empty nor degenerated.
346  /// Hence it contains a range of values.
347  [[nodiscard]] bool isProper() const noexcept {
348  return !this->isEmpty() && !this->isDegenerated();
349  }
350 
351  /// Returns true if region has no bounds.
352  [[nodiscard]] bool isUnbounded() const noexcept {
353  if (this->isEmpty()) {
354  return false;
355  }
356  return allTrue(this->min() == PointTraits<TPoint>::min()) &&
357  allTrue(this->max() == PointTraits<TPoint>::max());
358  }
359 
360  private:
361  explicit Region(UninitTag /*unused*/) {}
362 
363  // invariants:
364  // 1. min_==nullopt if and only if max_==nullopt
365  //
366  // 2. min_==nullopt or *min_ <= *max_
367  std::optional<TPoint> min_;
368  std::optional<TPoint> max_;
369 };
