88  /// Returns subview with shared ownership semantics of whole image.
89  [[nodiscard]] DynImageView subview(
90  Eigen::Vector2i uv, sophus::ImageSize size) const {
91  SOPHUS_ASSERT(imageSize().contains(uv));
92  SOPHUS_ASSERT_LE(uv.x() + size.width, this->layout_.width());
93  SOPHUS_ASSERT_LE(uv.y() + size.height, this->layout_.height());
94 
95  auto const layout =
96  ImageLayout::makeFromSizeAndPitchUnchecked(size, pitchBytes());
97  const size_t row_offset =
98  uv.x() * numBytesPerPixelChannel() * numChannels();
99  uint8_t const* ptr = this->rawPtr() + uv.y() * pitchBytes() + row_offset;
100  return DynImageView{layout, this->pixel_format_, ptr};
101  }
