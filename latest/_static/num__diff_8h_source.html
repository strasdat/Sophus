23 template <class TScalar>
24 class Curve {
25  public:
26  template <class TFn>
27  static auto numDiff(TFn curve, TScalar t, TScalar h) -> decltype(curve(t)) {
28  using ReturnType = decltype(curve(t));
29  static_assert(
30  std::is_floating_point<TScalar>::value,
31  "Scalar must be a floating point type.");
32  static_assert(
33  IsFloatingPoint<ReturnType>::kValue,
34  "ReturnType must be either a floating point scalar, "
35  "vector or matrix.");
36 
37  return (curve(t + h) - curve(t - h)) / (TScalar(2) * h);
38  }
39 };

41 template <class TScalar, int kMatrixDim, int kM>
42 class VectorField {
43  public:
44  static Eigen::Matrix<TScalar, kMatrixDim, kM> numDiff(
45  std::function<Eigen::Vector<TScalar, kMatrixDim>(
46  Eigen::Vector<TScalar, kM>)> vector_field,
47  Eigen::Vector<TScalar, kM> const& a,
48  TScalar eps) {
49  static_assert(
50  std::is_floating_point<TScalar>::value,
51  "Scalar must be a floating point type.");
52  Eigen::Matrix<TScalar, kMatrixDim, kM> j;
53  Eigen::Vector<TScalar, kM> h;
54  h.setZero();
55  for (int i = 0; i < kM; ++i) {
56  h[i] = eps;
57  j.col(i) =
58  (vector_field(a + h) - vector_field(a - h)) / (TScalar(2) * eps);
59  h[i] = TScalar(0);
60  }
61 
62  return j;
63  }
64 };
