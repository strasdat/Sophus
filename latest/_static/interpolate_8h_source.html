20 /// This function interpolates between two Lie group elements
21 /// ``foo_transform_bar`` and ``foo_transform_daz`` with an interpolation factor
22 /// of ``alpha`` in [0, 1].
23 ///
24 /// It returns a pose ``foo_T_quiz`` with ``quiz`` being a frame between ``bar``
25 /// and ``baz``. If ``alpha=0`` it returns ``foo_transform_bar``. If it is 1, it
26 /// returns ``foo_transform_daz``.
27 ///
28 /// (Since interpolation on Lie groups is inverse-invariant, we can equivalently
29 /// think of the input arguments as being ``bar_transform_foo``,
30 /// ``baz_transform_foo`` and the return value being ``quiz_T_foo``.)
31 ///
32 /// Precondition: ``p`` must be in [0, 1].
33 ///
34 template <class TGroup, class TScalar2 = typename TGroup::Scalar>
35 std::enable_if_t<interp_details::Traits<TGroup>::kSupported, TGroup>
36 interpolate(
37  TGroup const& foo_transform_bar,
38  TGroup const& foo_transform_daz,
39  TScalar2 p = TScalar2(0.5f)) {
40  using Scalar = typename TGroup::Scalar;
41  Scalar inter_p(p);
42  SOPHUS_ASSERT(
43  inter_p >= Scalar(0) && inter_p <= Scalar(1),
44  "p ({}) must in [0, 1].",
45  inter_p);
46  return foo_transform_bar *
47  TGroup::exp(
48  inter_p * (foo_transform_bar.inverse() * foo_transform_daz).log());
49 }
