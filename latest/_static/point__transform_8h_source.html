17 /// Projects 3-point (a,b,psi) = (x/z,y/z,1/z) through the origin (0,0,0) onto
18 /// the plane z=1. Hence it returns (a,b) = (x/z, y/z).
19 template <class TT>
20 Eigen::Matrix<TT, 2, 1> proj(
21  InverseDepthPoint3<TT> const& inverse_depth_point) {
22  return inverse_depth_point.projInZ1Plane();

36 /// Returns pose derivative of inverse depth point projection at the identity:
37 ///
38 /// Dx proj(exp(x) * y) at x=0
39 ///
40 /// with y = (a,b,psi) being an inverse depth point.
41 template <class TT>
42 Eigen::Matrix<TT, 2, 6> dxProjExpXPointAt0(
43  InverseDepthPoint3<TT> const& inverse_depth_point) {
44  Eigen::Matrix<TT, 2, 6> dx;
45  TT i = TT(1);
46  TT psi = inverse_depth_point.psi();
47 
48  TT a = inverse_depth_point.projInZ1Plane().x();
49  TT b = inverse_depth_point.projInZ1Plane().y();
50  // clang-format off
51  dx <<
52  psi, 0, -psi * a, -a*b, a*a + i, -b,
53  0, psi, -psi * b, -b*b - i, a*b, a;
54  // clang-format on
55  return dx;
56 }

58 /// Transforms inverse_depth point in frame bar to a scaled inverse depth point
59 /// in frame foo. Here the scale is psi, the input inverse depth.
60 ///
61 /// Given (a,b,psi) being the inverse depth point in frame bar, it returns
62 ///
63 /// psi * (foo_from_bar * inverse_depth_point_in_bar.toEuclideanPoint3())
64 ///
65 /// for psi!=0.
66 template <class TT>
67 Eigen::Matrix<TT, 3, 1> scaledTransform(
68  sophus::Se3<TT> const& foo_from_bar,
69  InverseDepthPoint3<TT> const& inverse_depth_point_in_bar) {
70  return foo_from_bar.so3() *
71  unproj(inverse_depth_point_in_bar.projInZ1Plane()) +
72  inverse_depth_point_in_bar.psi() * foo_from_bar.translation();
73 }

75 /// Transforms inverse_depth point from frame bar to frame foo followed by a
76 /// projection.
77 ///
78 /// If psi != 0, hence the point is not at +/- infinity, this function is
79 /// equivalent to:
80 ///
81 /// camProj(foo_from_bar * inverse_depth_point_in_bar.toEuclideanPoint3());
82 ///
83 /// However, this function can also applied when 1/z==0, hence the point is at
84 /// +/- infinity.
85 template <class TT>
86 Eigen::Matrix<TT, 2, 1> projTransform(
87  sophus::Se3<TT> const& foo_from_bar,
88  InverseDepthPoint3<TT> const& inverse_depth_point_in_bar) {
89  // R * (x,y,z) + t
90  // = z * [R * (x/z, y/z, 1) + 1/z * t]
91  //
92  // Hence:
93  // proj(R * (x,y,z) + t)
94  // = proj(1/z * [R * (x,y,z) + t]) { since proj(xyz) == proj(l * xyz) }
95  // = proj(R * (x/z, y/z, 1) + 1/z * t)
96  //
97  // qed.
98  // with R := foo_from_bar.so3(),
99  // t := foo_from_bar.translation()
100  // (x/z, y/z, 1) := unproj(rojInZ1Plane())
101  // 1/z := psi()
102  //
103  return proj(scaledTransform(foo_from_bar, inverse_depth_point_in_bar));
104 }

106 /// Functor to efficiently transform a number of point given a Se3 pose.
107 ///
108 /// When transforming a point `point_in_bar` given a sophus::Se3 pose
109 /// `foo_from_bar`, one can simply use
110 ///
111 /// ``Eigen::Vector3d = foo_from_bar * point_in_bar;``
112 ///
113 /// Internally, this applies the (unit) quaternion to the left and the right of
114 /// the point to rotate it and then adds the translation:
115 ///
116 /// point_in_foo = q*point_in_bar*q' + bar_origin_in_foo.
117 ///
118 /// If there are a lot of point to transform, there is a more efficient
119 /// way using the rotation matrix R which can be precomputed from the quaternion
120 /// q.
121 ///
122 /// point_in_foo = R * point_in_bar + bar_origin_in_foo
123 ///
124 /// This is what this functor is for.
125 template <class TT>
126 class PointTransformer {
127  public:
128  PointTransformer() = default;
129  explicit PointTransformer(sophus::Se3<TT> const& foo_from_bar)
130  : foo_from_bar_(foo_from_bar),
131  foo_rotation_bar_(foo_from_bar.so3().matrix()),
132  bar_origin_in_foo_(foo_from_bar.translation()) {}
133 
134  /// Transforms a 3-point from frame bar to frame foo.
135  [[nodiscard]] Eigen::Matrix<TT, 3, 1> transform(
136  Eigen::Matrix<TT, 3, 1> const& point_in_bar) const {
137  return foo_rotation_bar_ * point_in_bar + bar_origin_in_foo_;
138  }
139 
140  [[nodiscard]] Eigen::Matrix<TT, 3, 1> scaledTransform(
141  InverseDepthPoint3<TT> const& inverse_depth_point_in_bar) const {
142  return foo_rotation_bar_ *
143  unproj(inverse_depth_point_in_bar.projInZ1Plane()) +
144  inverse_depth_point_in_bar.psi() * bar_origin_in_foo_;
145  }
146 
147  /// Transforms 3-point in frame bar to foo and projects it onto the
148  /// Euclidean plane z=1 in foo.
149  [[nodiscard]] Eigen::Matrix<TT, 2, 1> projTransform(
150  Eigen::Matrix<TT, 3, 1> const& point_in_bar) const {
151  return proj(foo_rotation_bar_ * point_in_bar + bar_origin_in_foo_);
152  }
153 
154  /// Transforms and projects the 3d inverse depth point in frame bar to the
155  /// Euclidean plane z=1 in foo.
156  [[nodiscard]] Eigen::Matrix<TT, 2, 1> projTransform(
157  InverseDepthPoint3<TT> const& inverse_depth_point_in_bar) const {
158  return proj(scaledTransform(inverse_depth_point_in_bar));
159  }
160 
161  /// Returns pose derivative of inverse depth point projection at the identity:
162  ///
163  /// Dx proj(exp(x) * foo_from_bar * foo_in_bar.toEuclideanPoint3()) at x=0
164  ///
165  /// with foo_in_bar = (a,b,psi) being an inverse depth point.
166  [[nodiscard]] Eigen::Matrix<TT, 2, 6> dxProjExpXTransformPointAt0(
167  InverseDepthPoint3<TT> const& inverse_depth_point_in_bar) const {
168  Eigen::Matrix<TT, 2, 6> dx;
169  TT i = TT(1);
170  TT psi = inverse_depth_point_in_bar.psi();
171  Eigen::Matrix<TT, 3, 1> scaled_point_in_foo =
172  scaledTransform(inverse_depth_point_in_bar);
173  TT x = scaled_point_in_foo[0];
174  TT y = scaled_point_in_foo[1];
175  TT z = scaled_point_in_foo[2];
176  TT z_sq = z * z;
177 
178  // clang-format off
179  dx <<
180  psi/z, 0, -psi * x / z_sq, -x*y/z_sq, x*x/z_sq + i, -y/z,
181  0, psi/z, -psi * y / z_sq, -y*y/z_sq - i, x*y/z_sq, x/z;
182  // clang-format on
183  return dx;
184  }

186  // Assuming ProjectionZ1 based camera
187  [[nodiscard]] Eigen::Matrix<TT, 2, 3> dxProjTransformX(
188  InverseDepthPoint3<TT> const& inverse_depth_point_in_bar) const {
189  Eigen::Vector3<TT> const& r0 = this->fooRotationBar().col(0);
190  Eigen::Vector3<TT> const& r1 = this->fooRotationBar().col(1);
191  Eigen::Vector3<TT> const& t = this->barOriginInFoo();
192 
193  Eigen::Matrix3<TT> mat_j;
194  mat_j << r0, r1, t;
195 
196  return ProjectionZ1::dxProjX(scaledTransform(inverse_depth_point_in_bar)) *
197  mat_j;
198  }

200  [[nodiscard]] sophus::Se3<TT> const& fooFromBar() const {
201  return foo_from_bar_;
202  }
203  [[nodiscard]] Eigen::Matrix<TT, 3, 3> const& fooRotationBar() const {
204  return foo_rotation_bar_;
205  }
206  [[nodiscard]] Eigen::Matrix<TT, 3, 1> const& barOriginInFoo() const {
207  return bar_origin_in_foo_;
208  }
209 
210  private:
211  sophus::Se3<TT> foo_from_bar_;
212  Eigen::Matrix<TT, 3, 3> foo_rotation_bar_;
213  Eigen::Matrix<TT, 3, 1> bar_origin_in_foo_;
214 };
