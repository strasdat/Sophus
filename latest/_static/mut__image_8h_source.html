35  UniqueDataAreaDeleter(size_t num_bytes) : num_bytes(num_bytes) {}
36  void operator()(uint8_t* p) const {
37  if (p != nullptr) {
38  TAllocator().deallocate(p, num_bytes);

73 /// A image with write access to pixels and exclusive ownership. There is no
74 /// copy constr / copy assignment, but move constr / assignment.
75 ///
76 /// Content from a MutImage can be moved into an Image.
77 ///
78 /// Type is nullable. In that case `this->isEmpty()` is true.
79 ///
80 template <class TPixel, class TAllocator = Eigen::aligned_allocator<uint8_t>>
81 class MutImage : public MutImageView<TPixel> {
82  public:
83  template <class TT, class TAllocator2T>
84  friend class Image;
85 
86  template <class TPredicate, class TAllocator2T>
87  friend class MutRuntimeImage;
88 
89  /// Constructs empty image.
90  MutImage() : unique_(nullptr) {}
91 
92  /// Creates new image with given shape.
93  ///
94  /// If shape is not empty, memory allocation will happen.
95  explicit MutImage(ImageShape shape)
96  : MutImageView<TPixel>(shape, nullptr), unique_(nullptr) {
97  if (shape.sizeBytes() != 0u) {
98  SOPHUS_ASSERT_EQ(shape.sizeBytes() % sizeof(TPixel), 0);
99  this->unique_ = UniqueDataArea<TAllocator>(
100  TAllocator().allocate(shape.sizeBytes()),
101  MaybeLeakingUniqueDataAreaDeleter<TAllocator>(
102  UniqueDataAreaDeleter<TAllocator>(shape.sizeBytes())));
103  }
104  this->ptr_ = reinterpret_cast<TPixel*>(this->unique_.get());
105  }
106 
107  /// Creates new contiguous image with given size.
108  ///
109  /// If shape is not empty, memory allocation will happen.
110  explicit MutImage(sophus::ImageSize size)
111  : MutImage(ImageShape::makeFromSize<TPixel>(size)) {}
112 
113  /// Creates contiguous copy from view.
114  ///
115  /// If view is not empty, memory allocation will happen.
116  [[nodiscard]] static MutImage makeCopyFrom(ImageView<TPixel> const& view) {
117  MutImage image(view.imageSize());
118  image.copyDataFrom(view);
119  return image;
120  }
121 
122  /// Creates new MutImage given view and unary transform function.
123  ///
124  /// mut_image(u, v) = unary_op(view(u, v));
125  template <class TOtherPixel, class TUnaryOperation>
126  static MutImage makeFromTransform(
127  ImageView<TOtherPixel> view, TUnaryOperation const& unary_op) {
128  MutImage mut_image(view.imageSize());
129  mut_image.transformFrom(view, unary_op);
130  return mut_image;
131  }
132 
133  /// Creates new MutImage given two views and binary transform function.
134  ///
135  /// mut_image(u, v) = binary_op(lhs(u, v), rhs(u, v));
136  template <class TLhsPixel, class TRhsPixel, class TBinaryOperation>
137  static MutImage makeFromTransform(
138  ImageView<TLhsPixel> lhs,
139  ImageView<TRhsPixel> rhs,
140  TBinaryOperation const& binary_op) {
141  MutImage mut_image(lhs.imageSize());
142  mut_image.transformFrom(lhs, rhs, binary_op);
143  return mut_image;
144  }
145 
146  // Begin(Rule of 5):
147  // https://en.cppreference.com/w/cpp/language/rule_of_three
148 
149  /// Destructor
150  ~MutImage() { reset(); }
151 
152  /// Not copy constructable
153  MutImage(MutImage<TPixel> const& other) = delete;
154 
155  /// Not copy assignable
156  MutImage& operator=(MutImage const&) = delete;
157 
158  /// Move constructor - is cheap - no memory allocations.
159  MutImage(MutImage&& img) noexcept
160  : MutImageView<TPixel>(img.viewMut()), unique_(std::move(img.unique_)) {
161  this->ptr_ = reinterpret_cast<TPixel*>(unique_.get());
162  img.setViewToEmpty();
163  }
164 
165  /// Move assignment
166  MutImage& operator=(MutImage&& img) noexcept {
167  reset();
168  this->shape_ = img.shape_;
169  this->unique_ = std::move(img.unique_);
170  this->ptr_ = reinterpret_cast<TPixel*>(unique_.get());
171  img.setViewToEmpty();
172  return *this;
173  }
174  // End (Rule of 5)
175 
176  [[nodiscard]] MutImageView<TPixel> viewMut() const {
177  return MutImageView<TPixel>(this->shape(), this->ptrMut());
178  }
179 
180  /// Swaps img and this.
181  void swap(MutImage& img) {
182  std::swap(img.shape_, this->shape_);
183  std::swap(img.ptr_, this->ptr_);
184  std::swap(img.unique_, this->unique_);
185  }
186 
187  /// Clears image.
188  ///
189  /// If image was not empty, memory deallocations will happen.
190  void reset() {
191  unique_.reset();
192  this->setViewToEmpty();
193  }
194 
195  protected:
196  /// Leaks memory and returns deleter.
197  MaybeLeakingUniqueDataAreaDeleter<TAllocator> leakAndReturnDeleter() {
198  SOPHUS_ASSERT(!this->isEmpty());
199  MaybeLeakingUniqueDataAreaDeleter<TAllocator>& del =
200  *std::get_deleter<MaybeLeakingUniqueDataAreaDeleter<TAllocator>>(
201  unique_);
202  MaybeLeakingUniqueDataAreaDeleter<TAllocator> del_copy = del;
203  del.image_deleter.reset();
204  return del_copy;
205  }
206 
207  /// MutImage has unique ownership.
208  UniqueDataArea<TAllocator> unique_;
209 };
