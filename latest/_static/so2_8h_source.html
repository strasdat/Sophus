65 /// So2 base type - implements So2 class but is storage agnostic.
66 ///
67 /// SO(2) is the group of rotations in 2d. As a matrix group, it is the set of
68 /// matrices which are orthogonal such that ``R * R' = I`` (with ``R'`` being
69 /// the transpose of ``R``) and have a positive determinant. In particular, the
70 /// determinant is 1. Let ``theta`` be the rotation angle, the rotation matrix
71 /// can be written in close form:
72 ///
73 /// | cos(theta) -sin(theta) |
74 /// | sin(theta) cos(theta) |
75 ///
76 /// As a matter of fact, the first column of those matrices is isomorph to the
77 /// set of unit complex numbers U(1). Thus, internally, So2 is represented as
78 /// complex number with length 1.
79 ///
80 /// SO(2) is a compact and commutative group. First it is compact since the set
81 /// of rotation matrices is a closed and bounded set. Second it is commutative
82 /// since ``R(alpha) * R(beta) = R(beta) * R(alpha)``, simply because ``alpha +
83 /// beta = beta + alpha`` with ``alpha`` and ``beta`` being rotation angles
84 /// (about the same axis).
85 ///
86 /// Class invariant: The 2-norm of ``unit_complex`` must be close to 1.
87 /// Technically speaking, it must hold that:
88 ///
89 /// ``|unit_complex().squaredNorm() - 1| <= Constants::epsilon()``.
90 template <class TDerived>
91 class So2Base {
92  public:
93  static int constexpr kOptions = Eigen::internal::traits<TDerived>::kOptions;
94  using Scalar = typename Eigen::internal::traits<TDerived>::Scalar;
95  using ComplexT = typename Eigen::internal::traits<TDerived>::ComplexType;
96  using ComplexTemporaryType = Eigen::Matrix<Scalar, 2, 1, kOptions>;

132  /// Adjoint transformation
133  ///
134  /// This function return the adjoint transformation ``Ad`` of the group
135  /// element ``A`` such that for all ``x`` it holds that
136  /// ``hat(Ad_A * x) = A * hat(x) A^{-1}``. See hat-operator below.
137  ///
138  /// It simply ``1``, since ``SO(2)`` is a commutative group.
139  ///

168  /// Logarithmic map
169  ///
170  /// Computes the logarithm, the inverse of the group exponential which maps
171  /// element of the group (rotation matrices) to elements of the tangent space
172  /// (rotation angles).
173  ///
174  /// To be specific, this function computes ``vee(logmat(.))`` with
175  /// ``logmat(.)`` being the matrix logarithm and ``vee(.)`` the vee-operator
176  /// of SO(2).
177  ///
178  SOPHUS_FUNC [[nodiscard]] Scalar log() const {
179  using std::atan2;
180  return atan2(unitComplex().y(), unitComplex().x());
181  }

183  /// It re-normalizes ``unit_complex`` to unit length.
184  ///
185  /// Note: Because of the class invariant, there is typically no need to call
186  /// this function directly.
187  ///
188  SOPHUS_FUNC void normalize() {
189  using std::hypot;
190  // Avoid under/overflows for higher precision
191  Scalar length = hypot(unitComplex().x(), unitComplex().y());
192  SOPHUS_ASSERT(
193  length >= kEpsilon<Scalar>,
194  "Complex number should not be close to zero!");
195  mutUnitComplex() /= length;
196  }

198  /// Returns 2x2 matrix representation of the instance.
199  ///
200  /// For SO(2), the matrix representation is an orthogonal matrix ``R`` with
201  /// ``det(R)=1``, thus the so-called "rotation matrix".
202  ///
203  SOPHUS_FUNC [[nodiscard]] Transformation matrix() const {
204  Scalar const& real = unitComplex().x();
205  Scalar const& imag = unitComplex().y();
206  Transformation mat_r;
207  // clang-format off
208  mat_r <<
209  real, -imag,
210  imag, real;
211  // clang-format on
212  return mat_r;
213  }

224  /// Group multiplication, which is rotation concatenation.
225  ///
226  template <class TOtherDerived>
227  SOPHUS_FUNC So2Product<TOtherDerived> operator*(
228  So2Base<TOtherDerived> const& other) const {
229  using ResultT = ReturnScalar<TOtherDerived>;
230  Scalar const lhs_real = unitComplex().x();
231  Scalar const lhs_imag = unitComplex().y();
232  typename TOtherDerived::Scalar const& rhs_real = other.unitComplex().x();
233  typename TOtherDerived::Scalar const& rhs_imag = other.unitComplex().y();
234  // complex multiplication
235  ResultT const result_real = lhs_real * rhs_real - lhs_imag * rhs_imag;
236  ResultT const result_imag = lhs_real * rhs_imag + lhs_imag * rhs_real;
237 
238  ResultT const squared_norm =
239  result_real * result_real + result_imag * result_imag;
240  // We can assume that the squared-norm is close to 1 since we deal with a
241  // unit complex number. Due to numerical precision issues, there might
242  // be a small drift after pose concatenation. Hence, we need to renormalizes
243  // the complex number here.
244  // Since squared-norm is close to 1, we do not need to calculate the costly
245  // square-root, but can use an approximation around 1 (see
246  // http://stackoverflow.com/a/12934750 for details).
247  if (squared_norm != ResultT(1.0)) {
248  ResultT const scale = ResultT(2.0) / (ResultT(1.0) + squared_norm);
249  return So2Product<TOtherDerived>(
250  result_real * scale, result_imag * scale);
251  }
252  return So2Product<TOtherDerived>(result_real, result_imag);
253  }

255  /// Group action on 2-points.
256  ///
257  /// This function rotates a 2 dimensional point ``p`` by the So2 element
258  /// ``bar_R_foo`` (= rotation matrix): ``p_bar = bar_R_foo * p_foo``.
259  ///
260  template <
261  typename TPointDerived,
262  typename = typename std::enable_if<
263  IsFixedSizeVector<TPointDerived, 2>::value>::type>
264  SOPHUS_FUNC PointProduct<TPointDerived> operator*(
265  Eigen::MatrixBase<TPointDerived> const& p) const {
266  Scalar const& real = unitComplex().x();
267  Scalar const& imag = unitComplex().y();
268  return PointProduct<TPointDerived>(
269  real * p[0] - imag * p[1], //
270  imag * p[0] + real * p[1]);
271  }
272 
273  /// Group action on homogeneous 2-points.
274  ///
275  /// This function rotates a homogeneous 2 dimensional point ``p`` by the So2
276  /// element ``bar_R_foo`` (= rotation matrix): ``p_bar = bar_R_foo * p_foo``.
277  ///
278  template <
279  typename THPointDerived,
280  typename = typename std::enable_if<
281  IsFixedSizeVector<THPointDerived, 3>::value>::type>
282  SOPHUS_FUNC HomogeneousPointProduct<THPointDerived> operator*(
283  Eigen::MatrixBase<THPointDerived> const& p) const {
284  Scalar const& real = unitComplex().x();
285  Scalar const& imag = unitComplex().y();
286  return HomogeneousPointProduct<THPointDerived>(
287  real * p[0] - imag * p[1], //
288  imag * p[0] + real * p[1], //
289  p[2]);
290  }

292  /// Group action on lines.
293  ///
294  /// This function rotates a parametrized line ``l(t) = o + t * d`` by the So2
295  /// element:
296  ///
297  /// Both direction ``d`` and origin ``o`` are rotated as a 2 dimensional point
298  ///
299  SOPHUS_FUNC Line operator*(Line const& l) const {
300  return Line((*this) * l.origin(), (*this) * l.direction());
301  }
302 
303  /// Group action on hyper-planes.
304  ///
305  /// This function rotates a hyper-plane ``n.x + d = 0`` by the So2
306  /// element:
307  ///
308  /// Normal vector ``n`` is rotated
309  /// Offset ``d`` is left unchanged
310  ///
311  /// Note that in 2d-case hyper-planes are just another parametrization of
312  /// lines
313  ///
314  SOPHUS_FUNC Hyperplane operator*(Hyperplane const& p) const {
315  return Hyperplane((*this) * p.normal(), p.offset());
316  }

472  /// Group exponential
473  ///
474  /// This functions takes in an element of tangent space (= rotation angle
475  /// ``theta``) and returns the corresponding element of the group SO(2).
476  ///
477  /// To be more specific, this function computes ``expmat(hat(omega))``
478  /// with ``expmat(.)`` being the matrix exponential and ``hat(.)`` being the
479  /// hat()-operator of SO(2).
480  ///
481  SOPHUS_FUNC static So2<Scalar> exp(Tangent const& theta) {
482  using std::cos;
483  using std::sin;
484  return So2<Scalar>(cos(theta), sin(theta));
485  }

516  /// Returns the infinitesimal generators of SO(2).
517  ///
518  /// The infinitesimal generators of SO(2) is:
519  ///
520  /// | 0 -1 |
521  /// | 1 0 |
522  ///
523  SOPHUS_FUNC static Transformation generator() { return hat(Scalar(1)); }
524 
525  /// hat-operator
526  ///
527  /// It takes in the scalar representation ``theta`` (= rotation angle) and
528  /// returns the corresponding matrix representation of Lie algebra element.
529  ///
530  /// Formally, the hat()-operator of SO(2) is defined as
531  ///
532  /// ``hat(.): R^2 -> R^{2x2}, hat(theta) = theta * G``
533  ///
534  /// with ``G`` being the infinitesimal generator of SO(2).
535  ///
536  /// The corresponding inverse is the vee()-operator, see below.
537  ///
538  SOPHUS_FUNC static Transformation hat(Tangent const& theta) {
539  Transformation omega;
540  // clang-format off
541  omega <<
542  Scalar(0), -theta,
543  theta, Scalar(0);
544  // clang-format on
545  return omega;
546  }

549  ///

552  fitToSo2(Transformation const& r) {
553  return So2(makeRotationMatrix(r));
554  }

558  /// It returns the Lie bracket of SO(2). Since SO(2) is a commutative group,
559  /// the Lie bracket is simple ``0``.
560  ///
561  SOPHUS_FUNC static Tangent lieBracket(
562  Tangent const& /*unused*/, Tangent const& /*unused*/) {
563  return Scalar(0);
564  }

566  /// Draw uniform sample from SO(2) manifold.
567  ///
568  template <class TUniformRandomBitGenerator>
569  static So2 sampleUniform(TUniformRandomBitGenerator& generator) {
570  static_assert(
571  kIsUniformRandomBitGeneratorV<TUniformRandomBitGenerator>,
572  "generator must meet the UniformRandomBitGenerator concept");
573  std::uniform_real_distribution<Scalar> uniform(-kPi<Scalar>, kPi<Scalar>);
574  return So2(uniform(generator));
575  }

577  /// vee-operator
578  ///
579  /// It takes the 2x2-matrix representation ``Omega`` and maps it to the
580  /// corresponding scalar representation of Lie algebra.
581  ///
582  /// This is the inverse of the hat()-operator, see above.
583  ///
584  /// Precondition: ``Omega`` must have the following structure:
585  ///
586  /// | 0 -a |
587  /// | a 0 |
588  ///
589  SOPHUS_FUNC static Tangent vee(Transformation const& omega) {
590  using std::abs;
591  return omega(1, 0);
592  }
