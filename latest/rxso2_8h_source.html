54 /// RxSo2 base type - implements RxSo2 class but is storage agnostic
55 ///
56 /// This class implements the group ``R+ x SO(2)``, the direct product of the
57 /// group of positive scalar 2x2 matrices (= isomorph to the positive
58 /// real numbers) and the two-dimensional special orthogonal group SO(2).
59 /// Geometrically, it is the group of rotation and scaling in two dimensions.
60 /// As a matrix groups, R+ x SO(2) consists of matrices of the form ``s * R``
61 /// where ``R`` is an orthogonal matrix with ``det(R) = 1`` and ``s > 0``
62 /// being a positive real number. In particular, it has the following form:
63 ///
64 /// | s * cos(theta) s * -sin(theta) |
65 /// | s * sin(theta) s * cos(theta) |
66 ///
67 /// where ``theta`` being the rotation angle. Internally, it is represented by
68 /// the first column of the rotation matrix, or in other words by a non-zero
69 /// complex number.
70 ///
71 /// R+ x SO(2) is not compact, but a commutative group. First it is not compact
72 /// since the scale factor is not bound. Second it is commutative since
73 /// ``sR(alpha, s1) * sR(beta, s2) = sR(beta, s2) * sR(alpha, s1)``, simply
74 /// because ``alpha + beta = beta + alpha`` and ``s1 * s2 = s2 * s1`` with
75 /// ``alpha`` and ``beta`` being rotation angles and ``s1``, ``s2`` being scale
76 /// factors.
77 ///
78 /// This class has the explicit class invariant that the scale ``s`` is not
79 /// too close to either zero or infinity. Strictly speaking, it must hold that:
80 ///
81 /// ``complex().norm() >= Constants::epsilon()`` and
82 /// ``1. / complex().norm() >= Constants::epsilon()``.
83 ///
84 /// In order to obey this condition, group multiplication is implemented with
85 /// saturation such that a product always has a scale which is equal or greater
86 /// this threshold.
87 template <class TDerived>
88 class RxSo2Base {
89  public:

169  /// Returns group inverse.
170  ///
171  SOPHUS_FUNC [[nodiscard]] RxSo2<Scalar> inverse() const {
172  Scalar squared_scale = complex().squaredNorm();
173  Eigen::Vector2<Scalar> xy = complex() / squared_scale;
174  return RxSo2<Scalar>(xy.x(), -xy.y());
175  }

177  /// Logarithmic map
178  ///
179  /// Computes the logarithm, the inverse of the group exponential which maps
180  /// element of the group (scaled rotation matrices) to elements of the tangent
181  /// space (rotation-vector plus logarithm of scale factor).
182  ///
183  /// To be specific, this function computes ``vee(logmat(.))`` with
184  /// ``logmat(.)`` being the matrix logarithm and ``vee(.)`` the vee-operator
185  /// of RxSo2.
186  ///
187  SOPHUS_FUNC [[nodiscard]] Tangent log() const {
188  using std::log;
189  Tangent theta_sigma;
190  theta_sigma[1] = log(scale());
191  theta_sigma[0] = So2<Scalar>(complex()).log();
192  return theta_sigma;
193  }

195  /// Returns 2x2 matrix representation of the instance.
196  ///
197  /// For RxSo2, the matrix representation is an scaled orthogonal matrix ``sR``
198  /// with ``det(R)=s^2``, thus a scaled rotation matrix ``R`` with scale
199  /// ``s``.
200  ///
201  SOPHUS_FUNC [[nodiscard]] Transformation matrix() const {
202  Transformation s_r;
203  // clang-format off
204  s_r << complex()[0], -complex()[1],
205  complex()[1], complex()[0];
206  // clang-format on
207  return s_r;
208  }

219  /// Group multiplication, which is rotation concatenation and scale
220  /// multiplication.
221  ///
222  /// Note: This function performs saturation for products close to zero in
223  /// order to ensure the class invariant.
224  ///
225  template <class TOtherDerived>
226  SOPHUS_FUNC RxSo2Product<TOtherDerived> operator*(
227  RxSo2Base<TOtherDerived> const& other) const {
228  using ResultT = ReturnScalar<TOtherDerived>;
229 
230  Scalar lhs_real = complex().x();
231  Scalar lhs_imag = complex().y();
232  typename TOtherDerived::Scalar const& rhs_real = other.complex().x();
233  typename TOtherDerived::Scalar const& rhs_imag = other.complex().y();
234  /// complex multiplication
235  typename RxSo2Product<TOtherDerived>::ComplexType result_complex(
236  lhs_real * rhs_real - lhs_imag * rhs_imag,
237  lhs_real * rhs_imag + lhs_imag * rhs_real);
238 
239  const ResultT squared_scale = result_complex.squaredNorm();
240 
241  if (squared_scale < kEpsilon<ResultT> * kEpsilon<ResultT>) {
242  /// Saturation to ensure class invariant.
243  result_complex.normalize();
244  result_complex *= kEpsilonPlus<ResultT>;
245  }
246  if (squared_scale > Scalar(1.) / (kEpsilon<ResultT> * kEpsilon<ResultT>)) {
247  /// Saturation to ensure class invariant.
248  result_complex.normalize();
249  result_complex /= kEpsilonPlus<ResultT>;
250  }
251  return RxSo2Product<TOtherDerived>(result_complex);
252  }

254  /// Group action on 2-points.
255  ///
256  /// This function rotates a 2 dimensional point ``p`` by the So2 element
257  /// ``bar_R_foo`` (= rotation matrix) and scales it by the scale factor ``s``:
258  ///
259  /// ``p_bar = s * (bar_R_foo * p_foo)``.
260  ///
261  template <
262  typename TPointDerived,
263  typename = typename std::enable_if<
264  IsFixedSizeVector<TPointDerived, 2>::value>::type>
265  SOPHUS_FUNC PointProduct<TPointDerived> operator*(
266  Eigen::MatrixBase<TPointDerived> const& p) const {
267  return matrix() * p;
268  }

276  SOPHUS_FUNC HomogeneousPointProduct<THPointDerived> operator*(
277  Eigen::MatrixBase<THPointDerived> const& p) const {

282  /// Group action on lines.
283  ///
284  /// This function rotates a parameterized line ``l(t) = o + t * d`` by the So2
285  /// element and scales it by the scale factor
286  ///
287  /// Origin ``o`` is rotated and scaled
288  /// Direction ``d`` is rotated (preserving it's norm)
289  ///
290  SOPHUS_FUNC Line operator*(Line const& l) const {
291  return Line((*this) * l.origin(), (*this) * l.direction() / scale());
292  }
293 
294  /// Group action on hyper-planes.
295  ///
296  /// This function rotates a hyper-plane ``n.x + d = 0`` by the So2
297  /// element and scales offset by the scale factor
298  ///
299  /// Normal vector ``n`` is rotated
300  /// Offset ``d`` is scaled
301  ///
302  /// Note that in 2d-case hyper-planes are just another parametrization of
303  /// lines
304  ///
305  SOPHUS_FUNC Hyperplane operator*(Hyperplane const& p) const {
306  auto const this_scale = scale();
307  return Hyperplane(
308  (*this) * p.normal() / this_scale, this_scale * p.offset());
309  }

549  /// Returns derivative of exp(x) wrt. ``x``
550  ///
551  SOPHUS_FUNC static Eigen::Matrix<Scalar, kNumParameters, kDoF> dxExpX(
552  Tangent const& a) {
553  using std::cos;
554  using std::exp;
555  using std::sin;
556  Scalar const theta = a[0];
557  Scalar const sigma = a[1];
558

560  // clang-format off

596  /// Group exponential
597  ///
598  /// This functions takes in an element of tangent space (= rotation angle
599  /// plus logarithm of scale) and returns the corresponding element of the
600  /// group RxSo2.
601  ///
602  /// To be more specific, this function computes ``expmat(hat(theta))``
603  /// with ``expmat(.)`` being the matrix exponential and ``hat(.)`` being the
604  /// hat()-operator of RSO2.
605  ///
606  SOPHUS_FUNC static RxSo2<Scalar> exp(Tangent const& vec_a) {
607  using std::exp;
608  using std::max;
609  using std::min;
610 
611  Scalar const theta = vec_a[0];
612  Scalar const sigma = vec_a[1];
613  Scalar s = exp(sigma);
614  // Ensuring proper scale
615  s = max(s, kEpsilonPlus<Scalar>);
616  s = min(s, Scalar(1.) / kEpsilonPlus<Scalar>);
617  Eigen::Vector2<Scalar> z = So2<Scalar>::exp(theta).unitComplex();
618  z *= s;
619  return RxSo2<Scalar>(z);
620  }

622  /// Returns the ith infinitesimal generators of ``R+ x SO(2)``.
623  ///
624  /// The infinitesimal generators of RxSo2 are:
625  ///
626  /// ```
627  /// | 0 -1 |
628  /// G_0 = | 1 0 |
629  ///
630  /// | 1 0 |
631  /// G_1 = | 0 1 |
632  /// ```
633  ///
634  /// Precondition: ``i`` must be 0, or 1.
635  ///
636  SOPHUS_FUNC static Transformation generator(int i) {
637  SOPHUS_ASSERT(i >= 0 && i <= 1, "i should be 0 or 1.");
638  Tangent e;
639  e.setZero();
640  e[i] = Scalar(1);
641  return hat(e);
642  }
643 
644  /// hat-operator
645  ///
646  /// It takes in the 2-vector representation ``a`` (= rotation angle plus
647  /// logarithm of scale) and returns the corresponding matrix representation
648  /// of Lie algebra element.
649  ///
650  /// Formally, the hat()-operator of RxSo2 is defined as
651  ///
652  /// ``hat(.): R^2 -> R^{2x2}, hat(a) = sum_i a_i * G_i`` (for i=0,1,2)
653  ///
654  /// with ``G_i`` being the ith infinitesimal generator of RxSo2.
655  ///
656  /// The corresponding inverse is the vee()-operator, see below.
657  ///
658  SOPHUS_FUNC static Transformation hat(Tangent const& vec_a) {
659  Transformation mat_a;
660  // clang-format off
661  mat_a << vec_a(1), -vec_a(0),
662  vec_a(0), vec_a(1);
663  // clang-format on
664  return mat_a;
665  }
666 
667  /// Lie bracket
668  ///
669  /// It computes the Lie bracket of RxSO(2). To be more specific, it computes
670  ///
671  /// ``[omega_1, omega_2]_rxso2 := vee([hat(omega_1), hat(omega_2)])``
672  ///
673  /// with ``[A,B] := AB-BA`` being the matrix commutator, ``hat(.)`` the
674  /// hat()-operator and ``vee(.)`` the vee()-operator of RxSo2.
675  ///
676  SOPHUS_FUNC static Tangent lieBracket(
677  Tangent const& /*unused*/, Tangent const& /*unused*/) {
678  Eigen::Vector2<Scalar> res;
679  res.setZero();
680  return res;
681  }
682 
683  /// Draw uniform sample from RxSO(2) manifold.
684  ///
685  /// The scale factor is drawn uniformly in log2-space from [-1, 1],
686  /// hence the scale is in [0.5, 2)].
687  ///
688  template <class TUniformRandomBitGenerator>
689  static RxSo2 sampleUniform(TUniformRandomBitGenerator& generator) {
690  std::uniform_real_distribution<Scalar> uniform(Scalar(-1), Scalar(1));
691  using std::exp2;
692  return RxSo2(
693  exp2(uniform(generator)), So2<Scalar>::sampleUniform(generator));
694  }
695 
696  /// vee-operator
697  ///
698  /// It takes the 2x2-matrix representation ``Omega`` and maps it to the
699  /// corresponding vector representation of Lie algebra.
700  ///
701  /// This is the inverse of the hat()-operator, see above.
702  ///
703  /// Precondition: ``Omega`` must have the following structure:
704  ///
705  /// | d -x |
706  /// | x d |
707  ///
708  SOPHUS_FUNC static Tangent vee(Transformation const& omega) {
709  using std::abs;
710  return Tangent(omega(1, 0), omega(0, 0));
711  }
712 
713  protected:
714  SOPHUS_FUNC ComplexMember& mutComplex() { return complex_; }
715 
716  ComplexMember complex_; // NOLINT
717 };
