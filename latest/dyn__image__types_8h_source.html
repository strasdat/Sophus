57 // Call UserFunc with TDynImage cast to the appropriate concrete type
58 // from the options in pixelFormats...
59 template <class TUserFunc, typename TDynImage, typename... TpixelFormats>

62 // base case
63 template <class TUserFunc, typename TDynImage, typename TPixelFormat>
64 struct VisitImpl<TUserFunc, TDynImage, std::variant<TPixelFormat>> {
65  static void visit(TUserFunc&& func, TDynImage const& image) {
66  if (image.pixelFormat().template is<TPixelFormat>()) {
67  func(image.template imageView<TPixelFormat>());
68  }
69  }
70 };
71 
72 // inductive case
73 template <
74  typename TUserFunc,
75  typename TDynImage,
76  typename TPixelFormat,
77  typename... TRest>
78 struct VisitImpl<TUserFunc, TDynImage, std::variant<TPixelFormat, TRest...>> {
79  static void visit(TUserFunc&& func, TDynImage const& image) {
80  if (image.pixelFormat().template is<TPixelFormat>()) {
81  func(image.template imageView<TPixelFormat>());
82  } else {
83  VisitImpl<TUserFunc, TDynImage, std::variant<TRest...>>::visit(
84  std::forward<TUserFunc>(func), image);
85  }
86  }
87 };
88 } // namespace detail
