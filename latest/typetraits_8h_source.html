25 /// Type trait used to distinguish mappable vector types from scalars
26 ///
27 /// We use this class to distinguish Eigen::Vector<Scalar, kMatrixDim> from
28 /// Scalar types in LieGroup<TScalar>::Tangent
29 ///
30 /// Primary use is mapping LieGroup::Tangent over raw data, with 2 options:
31 /// - LieGroup::Tangent is "scalar" (for So2), then we just dereference pointer
32 /// - LieGroup::Tangent is Eigen::Vector<...>, then we need to use Eigen::Map
33 ///
34 /// Specialization of Eigen::internal::traits<TScalar> for TScalar is crucial
35 /// for for constructing Eigen::Map<TScalar>, thus we use that property for
36 /// distinguishing between those two options. At this moment there seem to be no
37 /// option to check this using only "external" API of Eigen
38 template <class TScalar>
39 using IsMappable =
40  IsSpecialized<Eigen::internal::traits<std::decay_t<TScalar>>>;
41 
42 template <class TScalar>
43 bool constexpr kIsMappableV = IsMappable<TScalar>::value;
