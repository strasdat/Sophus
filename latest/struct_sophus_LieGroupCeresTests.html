    StdVector<PointF64> point_vec;

        StdVector<LieGroupF64> const& group_vec,
        StdVector<PointF64> const& point_vec
        );

    // methods

    bool testAll();

        const size_t num_vertices,
        double const sigma_init,
        double const sigma_observation
        );
