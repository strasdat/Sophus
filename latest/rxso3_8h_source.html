54 /// RxSo3 base type - implements RxSo3 class but is storage agnostic
55 ///
56 /// This class implements the group ``R+ x SO(3)``, the direct product of the
57 /// group of positive scalar 3x3 matrices (= isomorph to the positive
58 /// real numbers) and the three-dimensional special orthogonal group SO(3).
59 /// Geometrically, it is the group of rotation and scaling in three dimensions.
60 /// As a matrix groups, RxSo3 consists of matrices of the form ``s * R``
61 /// where ``R`` is an orthogonal matrix with ``det(R)=1`` and ``s > 0``
62 /// being a positive real number.
63 ///
64 /// Internally, RxSo3 is represented by the group of non-zero quaternions.
65 /// In particular, the scale equals the squared(!) norm of the quaternion ``q``,
66 /// ``s = |q|^2``. This is a most compact representation since the degrees of
67 /// freedom (kDoF) of RxSo3 (=4) equals the number of internal parameters (=4).
68 ///
69 /// This class has the explicit class invariant that the scale ``s`` is not
70 /// too close to either zero or infinity. Strictly speaking, it must hold that:
71 ///
72 /// ``quaternion().squaredNorm() >= Constants::epsilon()`` and
73 /// ``1. / quaternion().squaredNorm() >= Constants::epsilon()``.
74 ///
75 /// In order to obey this condition, group multiplication is implemented with
76 /// saturation such that a product always has a scale which is equal or greater
77 /// this threshold.
78 template <class TDerived>
79 class RxSo3Base {
80  public:
81  using Scalar = typename Eigen::internal::traits<TDerived>::Scalar;
82  using QuaternionType =
83  typename Eigen::internal::traits<TDerived>::QuaternionType;
84  using QuaternionTemporaryType = Eigen::Quaternion<Scalar>;

127  /// Adjoint transformation
128  ///
129  /// This function return the adjoint transformation ``Ad`` of the group
130  /// element ``A`` such that for all ``x`` it holds that
131  /// ``hat(Ad_A * x) = A * hat(x) A^{-1}``. See hat-operator below.
132  ///
133  /// For RxSO(3), it simply returns the rotation matrix corresponding to ``A``.
134  ///
135  SOPHUS_FUNC [[nodiscard]] Adjoint adj() const {
136  Adjoint res;
137  res.setIdentity();
138  res.template topLeftCorner<3, 3>() = rotationMatrix();
139  return res;
140  }

171  /// Logarithmic map
172  ///
173  /// Computes the logarithm, the inverse of the group exponential which maps
174  /// element of the group (scaled rotation matrices) to elements of the tangent
175  /// space (rotation-vector plus logarithm of scale factor).
176  ///
177  /// To be specific, this function computes ``vee(logmat(.))`` with
178  /// ``logmat(.)`` being the matrix logarithm and ``vee(.)`` the vee-operator
179  /// of RxSo3.
180  ///
181  SOPHUS_FUNC [[nodiscard]] Tangent log() const {
182  return logAndTheta().tangent;
183  }
184 
185  /// As above, but also returns ``theta = |omega|``.
186  ///
187  SOPHUS_FUNC [[nodiscard]] TangentAndTheta logAndTheta() const {
188  using std::log;
189 
190  Scalar scale = quaternion().squaredNorm();
191  TangentAndTheta result;
192  result.tangent[3] = log(scale);
193  auto omega_and_theta = So3<Scalar>(quaternion()).logAndTheta();
194  result.tangent.template head<3>() = omega_and_theta.tangent;
195  result.theta = omega_and_theta.theta;
196  return result;
197  }
198  /// Returns 3x3 matrix representation of the instance.
199  ///
200  /// For RxSo3, the matrix representation is an scaled orthogonal matrix ``sR``
201  /// with ``det(R)=s^3``, thus a scaled rotation matrix ``R`` with scale
202  /// ``s``.
203  ///
204  SOPHUS_FUNC [[nodiscard]] Transformation matrix() const {
205  Transformation s_r;
206 
207  Scalar const vx_sq = quaternion().vec().x() * quaternion().vec().x();
208  Scalar const vy_sq = quaternion().vec().y() * quaternion().vec().y();
209  Scalar const vz_sq = quaternion().vec().z() * quaternion().vec().z();
210  Scalar const w_sq = quaternion().w() * quaternion().w();
211  Scalar const two_vx = Scalar(2) * quaternion().vec().x();
212  Scalar const two_vy = Scalar(2) * quaternion().vec().y();
213  Scalar const two_vz = Scalar(2) * quaternion().vec().z();
214  Scalar const two_vx_vy = two_vx * quaternion().vec().y();
215  Scalar const two_vx_vz = two_vx * quaternion().vec().z();
216  Scalar const two_vx_w = two_vx * quaternion().w();
217  Scalar const two_vy_vz = two_vy * quaternion().vec().z();
218  Scalar const two_vy_w = two_vy * quaternion().w();
219  Scalar const two_vz_w = two_vz * quaternion().w();
220 
221  s_r(0, 0) = vx_sq - vy_sq - vz_sq + w_sq;
222  s_r(1, 0) = two_vx_vy + two_vz_w;
223  s_r(2, 0) = two_vx_vz - two_vy_w;
224 
225  s_r(0, 1) = two_vx_vy - two_vz_w;
226  s_r(1, 1) = -vx_sq + vy_sq - vz_sq + w_sq;
227  s_r(2, 1) = two_vx_w + two_vy_vz;
228 
229  s_r(0, 2) = two_vx_vz + two_vy_w;
230  s_r(1, 2) = -two_vx_w + two_vy_vz;
231  s_r(2, 2) = -vx_sq - vy_sq + vz_sq + w_sq;
232  return s_r;
233  }

244  /// Group multiplication, which is rotation concatenation and scale
245  /// multiplication.
246  ///
247  /// Note: This function performs saturation for products close to zero in
248  /// order to ensure the class invariant.
249  ///
250  template <class TOtherDerived>
251  SOPHUS_FUNC RxSo3Product<TOtherDerived> operator*(
252  RxSo3Base<TOtherDerived> const& other) const {
253  using std::sqrt;
254  using ResultT = ReturnScalar<TOtherDerived>;
255  using QuaternionProductType =
256  typename RxSo3Product<TOtherDerived>::QuaternionType;
257 
258  QuaternionProductType result_quaternion(
259  sophus::So3<double>::quaternionProduct<QuaternionProductType>(
260  quaternion(), other.quaternion()));
261 
262  ResultT scale = result_quaternion.squaredNorm();
263  if (scale < kEpsilon<ResultT>) {
264  SOPHUS_ASSERT(scale > ResultT(0), "Scale must be greater zero.");
265  /// Saturation to ensure class invariant.
266  result_quaternion.normalize();
267  result_quaternion.coeffs() *= sqrt(kEpsilonPlus<ResultT>);
268  }
269  if (scale > ResultT(1.) / kEpsilon<ResultT>) {
270  result_quaternion.normalize();
271  result_quaternion.coeffs() /= sqrt(kEpsilonPlus<ResultT>);
272  }
273  return RxSo3Product<TOtherDerived>(result_quaternion);
274  }

276  /// Group action on 3-points.
277  ///
278  /// This function rotates a 3 dimensional point ``p`` by the So3 element
279  /// ``bar_R_foo`` (= rotation matrix) and scales it by the scale factor
280  /// ``s``:
281  ///
282  /// ``p_bar = s * (bar_R_foo * p_foo)``.
283  ///
284  template <
285  typename TPointDerived,
286  typename = typename std::enable_if<
287  IsFixedSizeVector<TPointDerived, 3>::value>::type>
288  SOPHUS_FUNC PointProduct<TPointDerived> operator*(
289  Eigen::MatrixBase<TPointDerived> const& p) const {
290  // Follows http:///eigen.tuxfamily.org/bz/show_bug.cgi?id=459
291  Scalar scale = quaternion().squaredNorm();
292  PointProduct<TPointDerived> two_vec_cross_p = quaternion().vec().cross(p);
293  two_vec_cross_p += two_vec_cross_p;
294  return scale * p + (quaternion().w() * two_vec_cross_p +
295  quaternion().vec().cross(two_vec_cross_p));
296  }
297 
298  /// Group action on homogeneous 3-points. See above for more details.
299  ///
300  template <
301  typename THPointDerived,
302  typename = typename std::enable_if<
303  IsFixedSizeVector<THPointDerived, 4>::value>::type>
304  SOPHUS_FUNC HomogeneousPointProduct<THPointDerived> operator*(
305  Eigen::MatrixBase<THPointDerived> const& p) const {
306  auto const rsp = *this * p.template head<3>();
307  return HomogeneousPointProduct<THPointDerived>(
308  rsp(0), rsp(1), rsp(2), p(3));
309  }

311  /// Group action on lines.
312  ///
313  /// This function rotates a parametrized line ``l(t) = o + t * d`` by the So3
314  /// element ans scales it by the scale factor:
315  ///
316  /// Origin ``o`` is rotated and scaled
317  /// Direction ``d`` is rotated (preserving it's norm)
318  ///
319  SOPHUS_FUNC Line operator*(Line const& l) const {
320  return Line(
321  (*this) * l.origin(),
322  (*this) * l.direction() / quaternion().squaredNorm());
323  }

325  /// Group action on planes.
326  ///
327  /// This function rotates parametrized plane
328  /// ``n.x + d = 0`` by the So3 element and scales it by the scale factor:
329  ///
330  /// Normal vector ``n`` is rotated
331  /// Offset ``d`` is scaled
332  ///
333  SOPHUS_FUNC Hyperplane operator*(Hyperplane const& p) const {
334  auto const this_scale = scale();
335  return Hyperplane(
336  (*this) * p.normal() / this_scale, this_scale * p.offset());

419  /// Setter of quaternion using scaled rotation matrix ``sR``.
420  ///
421  /// Precondition: The 3x3 matrix must be "scaled orthogonal"
422  /// and have a positive determinant.
423  ///
424  SOPHUS_FUNC void setScaledRotationMatrix(Transformation const& s_r) {
425  Transformation squared_s_r = s_r * s_r.transpose();
426  Scalar squared_scale =
427  Scalar(1. / 3.) *
428  (squared_s_r(0, 0) + squared_s_r(1, 1) + squared_s_r(2, 2));
429  SOPHUS_ASSERT(
430  squared_scale >= kEpsilon<Scalar> * kEpsilon<Scalar>,
431  "Scale factor must be greater-equal epsilon.");
432  SOPHUS_ASSERT(
433  squared_scale < Scalar(1.) / (kEpsilon<Scalar> * kEpsilon<Scalar>),
434  "Inverse scale factor must be greater-equal epsilon.");
435  Scalar scale = sqrt(squared_scale);
436  quaternionNonconst() = s_r / scale;
437  quaternionNonconst().coeffs() *= sqrt(scale);
438  }

453  /// Returns derivative of this * RxSo3::exp(x) wrt. x at x=0
454  ///
455  SOPHUS_FUNC [[nodiscard]] Eigen::Matrix<Scalar, kNumParameters, kDoF>
456  dxThisMulExpXAt0() const {
457  Eigen::Matrix<Scalar, kNumParameters, kDoF> j;
458  Eigen::Quaternion<Scalar> const q = quaternion();
459  j.col(3) = q.coeffs() * Scalar(0.5);
460  Scalar const c0 = Scalar(0.5) * q.w();
461  Scalar const c1 = Scalar(0.5) * q.z();
462  Scalar const c2 = -c1;
463  Scalar const c3 = Scalar(0.5) * q.y();
464  Scalar const c4 = Scalar(0.5) * q.x();
465  Scalar const c5 = -c4;
466  Scalar const c6 = -c3;
467  j(0, 0) = c0;
468  j(0, 1) = c2;
469  j(0, 2) = c3;
470  j(1, 0) = c1;
471  j(1, 1) = c0;
472  j(1, 2) = c5;
473  j(2, 0) = c6;
474  j(2, 1) = c4;
475  j(2, 2) = c0;
476  j(3, 0) = c5;
477  j(3, 1) = c6;
478  j(3, 2) = c2;
479  return j;
480  }
481 
482  /// Returns derivative of log(this^{-1} * x) by x at x=this.
483  ///
484  SOPHUS_FUNC [[nodiscard]] Eigen::Matrix<Scalar, kDoF, kNumParameters>
485  dxLogThisInvTimesXAtThis() const {
486  auto& q = quaternion();
487  Eigen::Matrix<Scalar, kDoF, kNumParameters> j;
488  // clang-format off
489  j << q.w(), q.z(), -q.y(), -q.x(),
490  -q.z(), q.w(), q.x(), -q.y(),
491  q.y(), -q.x(), q.w(), -q.z(),
492  q.x(), q.y(), q.z(), q.w();
493  // clang-format on
494  const Scalar scaler = Scalar(2.) / q.squaredNorm();
495  return j * scaler;
496  }

638  /// Returns derivative of exp(x) wrt. x.
639  ///
640  SOPHUS_FUNC static Eigen::Matrix<Scalar, kNumParameters, kDoF> dxExpX(
641  Tangent const& a) {
642  using std::exp;
643  using std::sqrt;
644  Eigen::Matrix<Scalar, kNumParameters, kDoF> j;
645  Eigen::Vector3<Scalar> const omega = a.template head<3>();
646  Scalar const sigma = a[3];
647  Eigen::Quaternion<Scalar> quat = So3<Scalar>::exp(omega).unitQuaternion();
648  Scalar const scale = sqrt(exp(sigma));
649  Scalar const scale_half = scale * Scalar(0.5);
650 
651  j.template block<4, 3>(0, 0) = So3<Scalar>::dxExpX(omega) * scale;
652  j.col(3) = quat.coeffs() * scale_half;
653  return j;
654  }

665  /// Returns derivative of ``expmat(x)`` wrt. ``x_i at x=0``, with
666  /// ``expmat(.)`` being the matrix exponential.

671  /// Group exponential
672  ///
673  /// This functions takes in an element of tangent space (= rotation 3-vector
674  /// plus logarithm of scale) and returns the corresponding element of the
675  /// group RxSo3.
676  ///
677  /// To be more specific, thixs function computes ``expmat(hat(omega))``
678  /// with ``expmat(.)`` being the matrix exponential and ``hat(.)`` being the
679  /// hat()-operator of RSO3.
680  ///
681  SOPHUS_FUNC static RxSo3<Scalar> exp(Tangent const& a) {
682  return expAndTheta(a).rxso3;
683  }
684 
685  /// As above, but also returns ``theta = |omega|``.
686  ///
687  SOPHUS_FUNC static RxSo3AndTheta expAndTheta(Tangent const& vec_a) {
688  using std::exp;
689  using std::max;
690  using std::min;
691  using std::sqrt;
692 
693  RxSo3AndTheta rxso3_and_theta;
694 
695  Eigen::Vector3<Scalar> const vec_omega = vec_a.template head<3>();
696  Scalar sigma = vec_a[3];
697  Scalar scale = exp(sigma);
698  // Ensure that scale-factor contraint is always valid
699  scale = max(scale, kEpsilonPlus<Scalar>);
700  scale = min(scale, Scalar(1.) / kEpsilonPlus<Scalar>);
701  Scalar sqrt_scale = sqrt(scale);
702  auto so3_and_theta = So3<Scalar>::expAndTheta(vec_omega);
703  Eigen::Quaternion<Scalar> quat = so3_and_theta.so3.unitQuaternion();
704  quat.coeffs() *= sqrt_scale;
705  rxso3_and_theta.rxso3 = RxSo3<Scalar>(quat);
706  rxso3_and_theta.theta = so3_and_theta.theta;
707  return rxso3_and_theta;
708  }

710  /// Returns the ith infinitesimal generators of ``R+ x SO(3)``.
711  ///
712  /// The infinitesimal generators of RxSo3 are:
713  ///
714  /// ```
715  /// | 0 0 0 |
716  /// G_0 = | 0 0 -1 |
717  /// | 0 1 0 |
718  ///
719  /// | 0 0 1 |
720  /// G_1 = | 0 0 0 |
721  /// | -1 0 0 |
722  ///
723  /// | 0 -1 0 |
724  /// G_2 = | 1 0 0 |
725  /// | 0 0 0 |
726  ///
727  /// | 1 0 0 |
728  /// G_2 = | 0 1 0 |
729  /// | 0 0 1 |
730  /// ```
731  ///
732  /// Precondition: ``i`` must be 0, 1, 2 or 3.
733  ///
734  SOPHUS_FUNC static Transformation generator(int i) {
735  SOPHUS_ASSERT(i >= 0 && i <= 3, "i should be in range [0,3].");
736  Tangent e;
737  e.setZero();
738  e[i] = Scalar(1);
739  return hat(e);
740  }
741 
742  /// hat-operator
743  ///
744  /// It takes in the 4-vector representation ``a`` (= rotation vector plus
745  /// logarithm of scale) and returns the corresponding matrix representation
746  /// of Lie algebra element.
747  ///
748  /// Formally, the hat()-operator of RxSo3 is defined as
749  ///
750  /// ``hat(.): R^4 -> R^{3x3}, hat(a) = sum_i a_i * G_i`` (for i=0,1,2,3)
751  ///
752  /// with ``G_i`` being the ith infinitesimal generator of RxSo3.
753  ///
754  /// The corresponding inverse is the vee()-operator, see below.
755  ///
756  SOPHUS_FUNC static Transformation hat(Tangent const& a) {
757  Transformation mat_a;
758  // clang-format off
759  mat_a << a(3), -a(2), a(1),
760  a(2), a(3), -a(0),
761  -a(1), a(0), a(3);
762  // clang-format on
763  return mat_a;
764  }
765 
766  /// Lie bracket
767  ///
768  /// It computes the Lie bracket of RxSO(3). To be more specific, it computes
769  ///
770  /// ``[omega_1, omega_2]_rxso3 := vee([hat(omega_1), hat(omega_2)])``
771  ///
772  /// with ``[A,B] := AB-BA`` being the matrix commutator, ``hat(.)`` the
773  /// hat()-operator and ``vee(.)`` the vee()-operator of RxSo3.
774  ///
775  SOPHUS_FUNC static Tangent lieBracket(Tangent const& a, Tangent const& b) {
776  Eigen::Vector3<Scalar> const omega1 = a.template head<3>();
777  Eigen::Vector3<Scalar> const omega2 = b.template head<3>();
778  Eigen::Vector4<Scalar> res;
779  res.template head<3>() = omega1.cross(omega2);
780  res[3] = Scalar(0);
781  return res;
782  }
783 
784  /// Draw uniform sample from RxSO(3) manifold.
785  ///
786  /// The scale factor is drawn uniformly in log2-space from [-1, 1],
787  /// hence the scale is in [0.5, 2].
788  ///
789  template <class TUniformRandomBitGenerator>
790  static RxSo3 sampleUniform(TUniformRandomBitGenerator& generator) {
791  std::uniform_real_distribution<Scalar> uniform(Scalar(-1), Scalar(1));
792  using std::exp2;
793  return RxSo3(
794  exp2(uniform(generator)), So3<Scalar>::sampleUniform(generator));
795  }
796 
797  /// vee-operator
798  ///
799  /// It takes the 3x3-matrix representation ``Omega`` and maps it to the
800  /// corresponding vector representation of Lie algebra.
801  ///
802  /// This is the inverse of the hat()-operator, see above.
803  ///
804  /// Precondition: ``Omega`` must have the following structure:
805  ///
806  /// | d -c b |
807  /// | c d -a |
808  /// | -b a d |
809  ///
810  SOPHUS_FUNC static Tangent vee(Transformation const& omega) {
811  using std::abs;
812  return Tangent(omega(2, 1), omega(0, 2), omega(1, 0), omega(0, 0));
813  }
