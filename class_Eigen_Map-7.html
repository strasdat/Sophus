template class Eigen::Map<sophus::Sim3<TScalar>, kOptions>¶

Overview¶

Specialization of Eigen::Map for Sim3; derived from Sim3Base. More…

#include <sim3.h>

template <class TScalar, int kOptions>
class Map<sophus::Sim3<TScalar>, kOptions>: public sophus::Sim3Base {
public:
    // typedefs

    using Base = sophus::Sim3Base<Map<sophus::Sim3<TScalar>, kOptions>> ;
    using Scalar = TScalar ;
    using Transformation = typename Base::Transformation ;
    using Point = typename Base::Point ;
    using HomogeneousPoint = typename Base::HomogeneousPoint ;
    using Tangent = typename Base::Tangent ;
    using Adjoint = typename Base::Adjoint ;

    // construction

    Map(Scalar* coeffs);

    // methods

    SOPHUS_FUNC Map<sophus::RxSo3<Scalar>, kOptions>& rxso3();
    SOPHUS_FUNC Map<sophus::RxSo3<Scalar>, kOptions> const& rxso3() const;
    SOPHUS_FUNC Map<Eigen::Vector3<Scalar>, kOptions>& translation();
    SOPHUS_FUNC Map<Eigen::Vector3<Scalar>, kOptions> const& translation() const;

protected:
    // fields

    Map<sophus::RxSo3<Scalar>, kOptions> rxso3_;
    Map<Eigen::Vector3<Scalar>, kOptions> translation_;
};

Detailed Documentation¶

Specialization of Eigen::Map for Sim3; derived from Sim3Base.

Allows us to wrap Sim3 objects around POD array.
